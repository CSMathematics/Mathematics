\begin{multicols}{2}
\begin{rlist}
\item $ \log_{x}{4}=-2 $
\item $ \log_{x}{25}=-2 $
\item $ \log_{x}{\frac{1}{64}}=-3 $
\item $ \log_{x}{\frac{1}{49}}=-2 $
\item $ \log_{x}{\frac{1}{100}}=-2 $
\item $ \log_{x}{e^2}=-2 $
\end{rlist}
\end{multicols}
