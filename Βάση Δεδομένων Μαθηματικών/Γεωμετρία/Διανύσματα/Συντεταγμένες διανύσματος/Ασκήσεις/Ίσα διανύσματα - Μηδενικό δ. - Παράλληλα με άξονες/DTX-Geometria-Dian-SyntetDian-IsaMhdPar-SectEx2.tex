\begin{alist}
\item τα διανύσματα $ \vec{a},\vec{\beta} $ να είναι ίσα.
\item $ \vec{a}=\vec{0} $.
\item $ \vec{\beta}=\vec{0} $.
\item $\vec{a}\parallel x'x$ και $ \vec{a}\neq\vec{0} $.
\item $ \vec{\beta}\parallel y'y $ και $ \vec{\beta}\neq\vec{0} $.
\end{alist}
