\textbf{Ερώτημα Α}
\begin{rlist}
\item Η $f$ ορίζεται στο $D_f=\mathbb{R}$. Για κάθε $x\in\mathbb{R}$ έχουμε:
\begin{align*}
f'(x)&=\left(x^2e^{-x}\right)'=\left(x^2\right)'e^{-x}+x^2\left(e^{-x}\right)'=\\&=2xe^{-x}+x^2e^{-x}(-x)'=2xe^{-x}-x^2e^{-x}=e^{-x}(2x-x^2)
\end{align*}
\item Η $f$ ορίζεται στο $D_f=\mathbb{R}$. Για κάθε $x\in\mathbb{R}$ έχουμε:
\begin{align*}
f'(x)&=\left[\left(x^2-1\right)\ln{\left(x^2+1\right)}\right]'=\left(x^2-1\right)'\ln{\left(x^2+1\right)}+\left(x^2-1\right)\left[\ln{\left(x^2+1\right)}\right]'=\\&=
2x\ln{\left(x^2+1\right)}+\left(x^2-1\right)\frac{\left(x^2+1\right)'}{\left(x^2+1\right)^2}=2x\ln{\left(x^2+1\right)}+\left(x^2-1\right)\frac{2x}{\left(x^2+1\right)^2}
\end{align*}
\end{rlist}
