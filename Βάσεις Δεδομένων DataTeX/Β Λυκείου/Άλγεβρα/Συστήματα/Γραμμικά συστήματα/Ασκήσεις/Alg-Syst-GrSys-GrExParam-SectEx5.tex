\begin{alist}
\item να δείξετε ότι $ a=4 $ και $ \beta=2 $.
\item να σχεδιάσετε την ευθεία σε ορθογώνιο σύστημα συντεταγμένων.
\item να βρεθούν τα σημεία τομής της με τους άξονες $ x'x $ και $ y'y $
\end{alist}
