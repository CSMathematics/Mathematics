\paragraph{\large ΘΕΩΡΙΑ}\mbox{}\\
\begin{enumerate}[itemsep=5mm]
\item Πότε μια εξίσωση με μεταβλητές $ x,y $ ονομάζεται εξίσωση μιας καμπύλης $ C $?
\item Τι ονομάζεται συντελεστής διεύθυνσης μιας ευθείας $ \varepsilon $?
\item Πως ορίζεται ο συντελεστής διεύθυνσης μαις ευθείας $ \varepsilon $ που διέρχεται από δύο σημεία $ A\left(x_1,y_1 \right)  $ και $ B\left( x_2,y_2\right)  $?
\item Ποιά σχέση συνδέει τους συντελεστές διεύθυνσης $ \lambda_1, \lambda_2 $, εφόσον αυτοί ορίζονται, δύο ευθειών $ \varepsilon_1,\varepsilon_2 $ οι οποίες είναι μεταξύ τους 
\begin{multicols}{2}
\begin{enumerate}[label=\roman*.]
\item Παράλληλες
\item Κάθετες
\end{enumerate}
\end{multicols}
\item Πως ορίζεται η εξίσωση της ευθείας η οποία διέρχεται από γνωστό σημείο $ A\left(x_0,y_0 \right)  $, με συντελεστή διεύθυνσης $ \lambda $?
\end{enumerate}
\paragraph{\large ΑΣΚΗΣΕΙΣ}\mbox{}\\
\begin{enumerate}[itemsep=5mm]
	\item Να βρεθεί η γωνία που σχηματίζουν με τον άξονα $ x'x $ οι ευθείες που διέρχονται από τα σημεία 
\begin{multicols}{2}
	\begin{enumerate}[label=\roman*.]
	\item $ Α(2,4) $ και $ Β(-4,1) $
	\item $ Α(0,-2) $ και $ Β(\sqrt{3},-3) $
	\item $ Α(-\sqrt{2},3) $ και $ Β(0,4) $
	\item $ Α(3,2) $ και $ Β(7,2) $
	\end{enumerate}
\end{multicols}
\item Να βρεθεί η εξίσωση της ευθείας η οποία
\begin{enumerate}[itemsep=0mm,label=\roman*.]
	\item διέρχεται από το σημείο $ Α(1,2) $ και έχει συντελεστή διεύθυνσης $ \lambda=3 $.
	\item διέρχεται από το σημείο $ Α(-3,4) $ και έχει συντελεστή διεύθυνσης $ \lambda=-\frac{1}{2} $.
	\item διέρχεται από τα σημεία $ Α(2,-1) $ και $ Β(0,5) $.
	\item διέρχεται από το σημείο $ Α(-1,3) $ και είναι κάθετη στην ευθεία $ \varepsilon : y=-2x+5 $.
	\item διέρχεται από το σημείο $ Α(3,0) $ και είναι παράλληλη με την ευθεία $ \varepsilon : y=4x-2 $.
	\item σχηματίζει με τον άξονα $ x'x $ γωνία $ \omega=\frac{\pi}{4} $ και διέρχεται από το σημείο $ Α(-2,3) $.
	\item διέρχεται από το σημείο $ Α(-3,3) $ και είναι παράλληλη με το διάνυσμα $ \overrightarrow{\delta}=(4,2) $.
\end{enumerate}
\item Να βρεθεί η εξίσωση της ευθείας η οποία διέρχεται από το σημείο $ Α(1,2) $ και είναι :
\begin{enumerate}[itemsep=0mm,label=\roman*.]
	\item παράλληλη με τον άξονα $ x'x $.
	\item παράλληλη με τον άξονα $ y'y $.
	\item παράλληλη με την ευθεία $ x=3 $.
	\item παράλληλη με την ευθεία $ y=-4 $.
\end{enumerate}

\item Να βρεθούν οι εξισώσεις των πλευρών του τριγώνου $ AB\varGamma $ με $ A(-1,2),B(3,4) $ και $ \varGamma(4,1) $.
\item Δίνεται τρίγωνο $ AB\varGamma $ με $ Α(1,2),Β(5,5) $ και $ \varGamma(8,1) $. 
\begin{enumerate}[itemsep=0mm,label=\roman*.]
	\item Να βρεθούν οι εξισώσεις των πλευρών.
	\item Να δειχθεί οτι το τρίγωνο είναι ισοσκελές.
	\item Να δειχθεί οτι $ B=90\degree $.
\end{enumerate}
\item Δίνεται τρίγωνο $ AB\varGamma $ με $ A(-3,0), B(2,4) $ και $ \varGamma(5,-1) $. Να βρεθούν :
\begin{enumerate}[itemsep=0mm,label=\roman*.]
	\item Οι εξισώσεις των πλευρών του.
	\item Οι εξισώσεις των υψών $ \upsilon_a,\upsilon_\beta, \upsilon_\gamma $.
	\item Οι εξισώσεις των μεσοκαθέτων των πλευρών.
	\item Οι εξισώσεις των διαμέσων $ \mu_a, \mu_\beta, \mu_\gamma $.
\end{enumerate}
\item Δίνεται τρίγωνο $ AB\varGamma $ με $ A(3,1), B(5,2) $ ενώ $ \varepsilon : 2x+y-4=0 $ και $ \zeta : 2x-y-8=0 $ είναι οι εξισώσεις του ύψους $ \upsilon_\gamma $ και της πλευράς $ B\varGamma $ αντίστοιχα. Να βρεθούν
\begin{enumerate}[itemsep=0mm,label=\roman*.]
	\item Το σημείο $ \varGamma $.
	\item Το κέντρο βάρους $ G $ του τριγώνου.
	\item Οι εξισώσεις των πλευρών.
	\item Οι εξισώσεις των διαμέσων.
	\item Το κέντρο του περιγεγραμμένου κύκλου.
\end{enumerate}
\item Δίνεται τετράπλευρο $ AB\varGamma\varDelta $ με $ A(0,1), Β(2,-4), \varGamma(5,2) $ και $ \varDelta(4,4) $. Να βρεθούν 
\begin{enumerate}[itemsep=0mm,label=\roman*.]
	\item Οι εξισώσεις των διαγωνίων $ Α\varGamma $ και $ B\varDelta $.
	\item Το σημείο τομής $ M $ των διαγωνίων.
\end{enumerate}
\item Δίνεται τετράπλευρο $ AB\varGamma\varDelta $ με $ A(1,1), Β(0,-2), \varGamma(5,-3) $ ενώ $ M(3,-1) $ είναι το μέσο της $ B\varDelta $.
\begin{enumerate}[itemsep=0mm,label=\roman*.]
	\item Να βρεθεί η κορυφή $ \varDelta $ του τετραπλεύρου.
	\item Να δειχθεί οτι το τετράπλευρο είναι παραλληλόγραμμο.
\end{enumerate}
\item Δίνεται τετράπλευρο $ AB\varGamma\varDelta $ με $ A(0,2), Β(-1,-2), \varGamma(3,-1) $ και $ \varDelta(4,3) $.
\begin{enumerate}[itemsep=0mm,label=\roman*.]
	\item Να βρεθουν οι εξισώσεις των διαγωνίων του.
	\item Να δειχθεί οτι το τετράπλευρο είναι ρόμβος.
	\item Να βρεθεί το σημείο τομής $ M $ των διαγωνίων.
\end{enumerate}
\end{enumerate}
