itemsep=5mm]
\item \begin{enumerate}[label=\bf\roman*.]
	\item Να αποδειχθεί οτι το τετράγωνο μιας κάθετης πλευράς, ενός ορθογωνίου τριγώνου $ AB\varGamma $, ισούται με το γινόμενο της υποτείνουσας επί την προβολή της πλευράς αυτής στην υποτείνουσα.
	\item Να αποδειχθεί οτι σε ένα ορθογώνιο τρίγωνο $ AB\varGamma $, με $ A\varDelta $ ($ \upsilon_a $) το ύψος που αντιστοιχεί στην υποτείνουσα, ισχύουν οι παρακάτω σχέσεις.
\begin{multicols}{3}
\begin{enumerate}[label=\bf\alph*.]
\item $ a^2=\beta^2+\gamma^2 $
\item $ \frac{\beta^2}{\gamma^2}=\frac{\Gamma\varDelta}{B\varDelta} $
\item $ \frac{1}{\beta^2}+\frac{1}{\gamma^2}=\frac{1}{\upsilon_a^2} $
\end{enumerate}
\end{multicols}
\item Να χαρακτηριστούν οι παρακάτω προτάσεις ως σωστές (Σ) ή λανθασμένες (Λ).
\begin{enumerate}[label=\bf\alph*.]
\item Σε κάθε ορθογώνιο τρίγωνο με πλευρές $ a,\beta,\gamma $ ισχύει $ \frac{1}{\beta^2}+\frac{1}{\gamma^2}=\frac{1}{\upsilon_a^2} $.
\item Σε κάθε ορθογώνιο τρίγωνο το τετράγωνο μιας κάθετης πλευράς είναι ίσο με το γινόμενο της υποτείνουσας επί την προβολή της άλλης κάθετης στην υποτείνουσα.
\item Εαν $ AB\varGamma $ είναι ένα ορθογώνιο $ \left( A=90\degree\right)  $ και ισοσκελές τρίγωνο τότε θα ισχύει $ a=\sqrt{2}\beta $.
\item Το πηλίκο των τετραγώνων των κάθετων πλευρών ενός ορθογωνίου τριγώνου είναι ίσο με το πηλίκο των προβολών τους στην υποτείνουσα.
\end{enumerate}
\end{enumerate}
\item Δίνεται ημικύκλιο $ \widearc{AB} $ με διάμετρο το ευθύγραμμο τμήμα $ AB=10cm $. Έστω $ \varGamma $ ένα τυχαίο σημείο του ημικυκλίου. Αν $ Α\varGamma=6cm $ να βρεθούν οι πλευρές $ Β\varGamma, A\varDelta, B\varDelta $ και $ \varGamma\varDelta $.

\item Σε ισόπλευρο τρίγωνο $ AB\varGamma $ πλευράς $ a $, φέρουμε το ύψος $ A\varDelta $. Από το σημείο $ \varDelta $ σχεδιάζουμε κάθετες $ \varDelta E $ και $ \varDelta Z $ στις πλευρές $ AB $ και $ A\varGamma $ αντίστοιχα. Να δειχθεί οτι 
\begin{multicols}{2}
\begin{enumerate}[label=\bf\roman*.]
\item $ E\varGamma=\dfrac{a}{4} $.
\item $ \varDelta E=\dfrac{\sqrt{3}a}{4} $.
\item $ ZE=\dfrac{3a}{4} $.
\item $ AH=\dfrac{3\sqrt{3}a}{8} $.
\end{enumerate}
\end{multicols}
