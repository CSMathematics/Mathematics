\item 
\begin{enumerate}[label=\roman*.]
	\item Πότε μια εξίσωση με μεταβλητές $ x,y $ ονομάζεται εξίσωση μιας καμπύλης $ C $?
	\item Τι ονομάζεται συντελεστής διεύθυνσης μιας ευθείας $ \varepsilon $?
	\item Πως ορίζεται ο συντελεστής διεύθυνσης μιας ευθείας $ \varepsilon $ που διέρχεται από δύο σημεία $ A\left(x_1,y_1 \right)  $ και $ B\left( x_2,y_2\right)  $?
	\item Ποιά σχέση συνδέει τους συντελεστές διεύθυνσης $ \lambda_1, \lambda_2 $, εφόσον αυτοί ορίζονται, δύο ευθειών $ \varepsilon_1,\varepsilon_2 $ οι οποίες είναι μεταξύ τους 
	\begin{multicols}{2}
		\begin{enumerate}[label=\roman*.]
			\item Παράλληλες
			\item Κάθετες
		\end{enumerate}
	\end{multicols}
	\item Πως ορίζεται η εξίσωση της ευθείας η οποία διέρχεται από γνωστό σημείο $ A\left(x_0,y_0 \right)  $, με συντελεστή διεύθυνσης $ \lambda $?
\end{enumerate}\mbox{}\\\monades{20}
\item Να βρεθεί η εξίσωση της ευθείας η οποία :
\begin{enumerate}[label=\roman*.]
	\item διέρχεται από το σημείο $ Α(3,1) $ και είναι παράλληλη με το διάνυσμα $ \overrightarrow{\delta}=(4,2) $.
	\item διέρχεται από τα σημεία $ Α(-1,2) $ και $Β(2,5)$.
	\item διέρχεται από το σημείο $ Α(4,0) $ και σχηματίζει με τον άξονα $ x'x $ γωνία $ \varphi=\frac{\pi}{4} $.
	\item διέρχεται από το σημείο $ Α(3,4) $ και είναι κάθετη στην ευθεία $ y=2 $.
\end{enumerate}\monades{20}
\item Να βρεθεί ο γεωμετρικός τόπος των σημείων $ Μ(\lambda-3,2-\lambda) $ όταν το $ \lambda $ διατρέχει το σύνολο $ \mathbb{R} $ των πραγματικών αριθμών. Στη συνέχεια να βρεθεί το εμβαδόν του τριγώνου που σχηματίζει η ευθεία με τους άξονες $ x'x $ και $ y'y $.\\\monades{30}\\\\

\item Δίνεται ευθεία $ \varepsilon : x-2y+3=0 $ και ευθεία $ \zeta $ η οποία διέρχεται από το σημείο $ A\left( 0,\frac{7}{2}\right)  $ και είναι κάθετη στην $ \varepsilon $. Πάνω στην ευθεία $ \zeta $ παίρνουμε το σημείο $ B $ το οποίο είναι συμμετρικό του σημείου $ Α $ ως προς την ευθεία $ \varepsilon $.
\begin{enumerate}[label=\roman*.]
	\item Να βρεθεί η εξίσωση της ευθείας $ \zeta $.
	\item Να βρεθεί το σημείο τομής $ M $ των ευθειών $ \varepsilon $ και $ \zeta $.
	\item Να βρεθεί το σημείο $ Β $.
\end{enumerate}
