\paragraph{Προβλήματα}

\begin{center}
\begin{tikzpicture}
% Wall
\draw[pattern=north west lines] (-1,4) rectangle (1,3.5);
% Spring

\item Βρείτε το χρονικό διάστημα κατά το οποίο το σώμα απομακρύνεται από το έδαφος.
