\paragraph{\textcolor{black}{{\large ΘΕΜΑΤΑ}}}\mbox{}\\
\begin{enumerate}[label=\bf\textcolor{black}{{\large \arabic*.}},
itemsep=5mm]
\item Να απαντήσεις στις παρακάτω ερωτήσεις.
\begin{enumerate}[label=\roman*.]
\item Με ποιούς τρόπους μπορούμε να παραστήσουμε ένα σύνολο?
\item Πότε δύο σύνολα είναι ίσα?
\item Τι ονομάζουμε δειγματικό χώρο ενός πειράματος τύχης?
\item Πότε δύο ενδεχόμενα ονομάζονται ασυμβίβαστα?
\item Να διατυπωθεί ο κλασσικός ορισμός της πιθανότητας.
\end{enumerate}
\item Ρίχνουμε ένα ζάρι 2 φορές και καταγάφουμε τα αποτελέσματα.
\begin{enumerate}[label=\roman*.]
\item Να βρεθεί ο δειγματικός χώρος του πειράματος.
\item Να βρεθούν τα ενδεχόμενα
\begin{itemize}
\item $ Α=\left\lbrace\textrm{Το αποτέλεσμα της 1ης να είναι μεγαλύτερο από της 2ης.}\right\rbrace  $
\item $ Β=\left\lbrace\textrm{Το αποτέλεσμα των 2 ρίψεων να είναι ίδιο.}\right\rbrace  $
\item $ \varGamma=\left\lbrace\textrm{Το γινόμενο των 2 ρίψεων να είναι μεγαλύτερο από 15.}\right\rbrace  $
\end{itemize}
\item Να βρεθούν οι πιθανότητες των παραπάνω ενδεχομένων.
\end{enumerate}
\item Το $ 70\% $ μιας τάξης παίζει ποδόσφαιρο, το $ 40\% $ παίζει μπάσκετ και το $ 25\% $ παίζει και τα δύο αθλήματα.\\Αν επιλέξουμε τυχαία έναν μαθητή να βρεθούν οι πιθανότητες των ενδεχομένων :
\begin{enumerate}[label=\roman*.]
\item Ο μαθητής να παίζει τουλάχιστον ένα από τα δύο αθλήματα.
\item Να μην παιζεί κανένα από τα δύο αθλήματα.
\item Να μην παίζει ποδόσφαιρο.
\item Να μην παίζει μπάσκετ.
\item Να παίζει μόνο ποδόσφαιρο.
\item Να παίζει μόνο μπάσκετ.
\item Να παίζει ή μόνο ποδόσφαιρο ή μόνο μπάσκετ.
\end{enumerate}
\item Έστω Α,Β δύο ενδεχόμενα του ίδιου δειγματικού χώρου για τα οποία ισχύει $ P(A)=0.7 $ και $ P(B)=0.6 $.
\begin{enumerate}[label=\roman*.]
\item Να εξετάσεις αν τα ενδεχόμενα Α, Β είναι ασυμβίβαστα.
\item Να αποδειχθεί οτι $ P(A\cap B)\leq0.7 $ και $ P(A\cup B)\geq0.6 $.
\item Να αποδειχθεί οτι $ 0.3\leq P(A\cap B)\leq0.6 $.
\end{enumerate}
\end{enumerate}
