\paragraph{\textcolor{black}{{\large ΘΕΜΑΤΑ}}}\mbox{}\\
\begin{enumerate}[label=\bf\textcolor{black}{{\large \arabic*.}},
itemsep=5mm]
\item \begin{enumerate}[label=\bf\roman*.]
\item \begin{enumerate}[label=\bf\alph*.]
	\item Τι ονομάζουμε εξίσωση 2\textsuperscript{ου} βαθμού?
	\item Ποιός αριθμός μας δείχνει το πλήθος των ριζών μιας εξίσωσης 2\textsuperscript{ου} βαθμού?
	\item Πότε μια εξίσωση έχει 2 ρίζες, πότε μια και πότε είναι αδύνατη στο $ \mathbb{R} $?
	\item Να γραφούν οι τύποι του \selectlanguage{english}Vieta\selectlanguage{english}.
\end{enumerate}\monades{10}{.89}
\item Να χαρακτηριστούν οι παρακάτω εξισώσεις ως σωστές (Σ) ή λανθασμένες (Λ).
\begin{enumerate}[label=\bf\alph*.]
\item Αν για μια εξίσωση 2\textsuperscript{ου} βαθμού έχουμε $ \varDelta>0 $ τότε έχει 2 άνισες λύσεις.
\item Αν για μια εξίσωση 2\textsuperscript{ου} βαθμού έχουμε $ \varDelta<0 $ τότε έχει μια διπλή λύση.
\item Η εξίσωση $ ax^2+\beta x+\gamma=0 $ παριστάνει μια εξίσωση 2\textsuperscript{ου} βαθμού για κάθε τιμή του $ a $.
\item Αν $ x_1, x_2 $ είναι οι λύσεις μιας εξίσωσης 2\textsuperscript{ου} βαθμού τότε ισχύει : $ x_1+x_2=\dfrac{\beta}{a} $ και $ x_1\cdot x_2=\dfrac{\gamma}{a} $.
\item Αν $ x_1, x_2 $ είναι οι λύσεις μιας εξίσωσης 2\textsuperscript{ου} βαθμού με $ x_1=-x_2 $ τότε $ \beta=0 $.
\end{enumerate}
\end{enumerate}\monades{15}{.942}
\item Να λυθούν οι παρακάτω εξισώσεις 
\begin{multicols}{2}
\begin{enumerate}[label=\bf\roman*.]
	\item $ 3x^2-5x+7=1+2x^2 $
	\item $ 2x^2-5|x|+3=0 $\\
	\item $ \dfrac{x-1}{x}+\dfrac{2-x}{x-1}=\dfrac{1}{2} $
	\item $ \left( x+\dfrac{1}{x}\right)^2-3\left( x+\dfrac{1}{x}\right)+2=0  $
\end{enumerate}
\end{multicols}\monades{25}{.942}
\item Να δειχτεί οτι η εξίσωση
\[ ax^2+(a-1)x-1=0 \]
\begin{enumerate}[label=\bf\roman*.]
	\item έχει λύσεις για κάθε $ a\in\mathbb{R} $.\\\monades{15}{.89}
	\item έχει μια διπλή λύση για $ a=-1 $.\\\monades{10}{.89}
\end{enumerate}
\item Να βρεθούν οι σταθερές $ a, \beta\in\mathbb{R} $ έτσι ώστε η εξίσωση \[ ax^2+(2a-3\beta)x+(a-\beta+2)=0 \] να έχει λύσεις τις $ x_1=-2, x_2=1 .$\\\monades{25}{.942}
\end{enumerate}
