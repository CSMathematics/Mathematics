\begin{enumerate}[label=\bf\textcolor{black}{{\large \arabic*.}},
itemsep=5mm]
\item \begin{enumerate} \item[A.]Να απαντήσεις στις παρακάτω ερωτήσεις.
\begin{enumerate}[label=\roman*.]
\item Τι ονομάζεται εξίσωση 1ου βαθμού, ποιά μορφή έχει και πότε έχει μια μοναδική λύση?
\item Τι ονομάζουμε λύση μιας εξίσωσης?
\item Για ποιές τιμές του πραγματικού αριθμού $ a $, η εξίσωση $ ax^2+\beta x+\gamma=0 $, παριστάνει εξίσωση 2ου βαθμού?
\item Για ποιές τιμές της διακρίνουσας $ \varDelta $, η εξίσωση $ ax^2+\beta x+\gamma=0 $ έχει πραγματικές λύσεις?
\end{enumerate}
\item[B.] Να χαρακτηριστούν οι παρακάτω προτάσεις ως σωστές (Σ) ή λανθασμένες (Λ).
\begin{enumerate}[label=\roman*.]
\item Για $ \lambda=1 $ εξίσωση $ (\lambda-1)x=3 $ είναι αδύνατη.
\item Αν $ a>0 $ και $ \nu $ περιττός, τότε η εξίσωση $ x^\nu=a $ έχει μια λυση.
\item Αν $ a<0 $ και $ \nu $ περιττός, τότε η εξίσωση $ x^\nu=a $ έχει λυση την $ x=-\SQRT[\nu]{a} $.
\item Αν μια εξίσωση 2ου βαθμού έχει δύο άνισες αρνητικές ρίζες, τότε $ P<0 $.
\end{enumerate}
\end{enumerate}
\item Να λυθούν οι εξισώσεις :
\begin{enumerate}[label=\roman*.]
\item $ \left(x-3\right)^2-2(1-x)+4=x^2-4(x-3)-1 $
\item $ \lambda^2(x-1)=(2\lambda+3)x-1 $ για τις διάφορες τιμές του $ \lambda $.
\end{enumerate}
\item Να λυθεί η εξίσωση \[ |x-3|+2|x+1|=|4-x|+5 \]
\item Δίνεται τρίγωνο $ ΑΒ\varGamma $ με πλευρές $ a,\beta,\gamma $ και η εξίσωση \[ \left(\beta^2+\gamma^2\right)x^2-2a\gamma x+a^2-\beta^2=0 \]Να δειχθεί οτι αν η εξίσωση έχει μοναδική λύση τότε το τρίγωνο είναι ορθογώνιο.
\end{enumerate}
