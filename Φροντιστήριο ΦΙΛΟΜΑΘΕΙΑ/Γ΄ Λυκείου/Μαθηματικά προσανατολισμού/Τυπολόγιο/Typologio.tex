\item Μέγιστη και ελάχιστη τιμή μέτρου μιγαδικού, μέτρου διαφοράς μιγαδικών.\\
Αν $ Μ $ είναι η εικόνα ενός μιγαδικού $ z $ τότε για την ελάχιστη και μέγιστη τιμή του $ |z| $ ισχύει :
\begin{enumerate}[itemsep=0mm,label=\roman*.]
\item Αν η εικόνα του $ z $ κινείται σε ευθεία $ \varepsilon $ τότε : $ \min{|z|}=d(M,\varepsilon) $.
\item Αν η εικόνα του $ z $ κινείται πάνω σε κύκλο $ |z-z_0|=\rho $ με κέντρο $ M_0(z_0) $ τότε : \[ \min|z|=OM_0-\rho\;\;\textrm{ και }\;\;\max{|z|}=OM_0+\rho \] όπου $ O $ η αρχή των αξόνων.
\item Αν η εικόνα του $ z $ κινείται πάνω στην έλλειψη $ \dfrac{x^2}{a^2}+\dfrac{y^2}{\beta^2}=1 $ τότε : \[ \min{|z|}=\beta\;\;\textrm{ και }\;\;\max{|z|}=a \] όπου $ 2a,2\beta $ τα μήκη του μεγάλου και του μικρού άξονα της έλλειψης αντίστοιχα.
\item Αν η εικόνα του $ z $ κινείται πάνω στην υπερβολή $ \dfrac{x^2}{a^2}-\dfrac{y^2}{\beta^2}=1 $ τότε : \[ \min{|z|}=a \]
\end{enumerate}
Αν $ Μ_1,M_2 $ είναι οι εικόνες μιγαδικών $ z_1,z_2 $ τότε για την ελάχιστη και μέγιστη τιμή του μέτρου της διαφοράς τους $ |z_1-z_2| $ ισχύει :
\begin{enumerate}[itemsep=0mm,label=\roman*.]
\item Αν η εικόνες των $ z_1,z_2 $ κινoύνται σε παράλληλες ευθείες $ \varepsilon_1,\varepsilon_2 $ τότε : $ \min{|z_1-z_2|=d(\varepsilon_1,\varepsilon_2)} $.
\item Αν η εικόνες των $ z_1,z_2 $ κινoύνται σε κύκλο $ (Μ_0,\rho) $ τότε : $ \max{|z_1-z_2|=2\rho} $.
\item Αν η εικόνα του $ z_1 $ κινείται σε ευθεία $ \varepsilon $ και η εικόνα του $ z_2 $ σε κύκλο $ (Μ_0,\rho) $ τότε : \[ \min{|z_1-z_2|=d(M_0,\varepsilon)-\rho}\;\;\textrm{ και }\;\;\max{|z_1-z_2|=d(M_0,\varepsilon)+\rho} \]
\item Αν οι εικόνες των $ z_1,z_2 $ κινούνται σε κύκλους $ C_1 : (K_1,\rho_1) $ και $ C_2 : (K_2,\rho_2) $ αντίστοιχα τότε :
\begin{enumerate}[itemsep=0mm,label=\alph*.]
\item Αν οι κύκλοι δεν έχουν κοινά σημεία και βρίσκεται εξωτερικά ο ένας του άλλου δηλαδή $ K_1K_2>\rho_1+\rho_2 $ τότε : \[ \min{|z_1-z_2|=K_1K_2-\rho_1-\rho_2}\;\;\textrm{ και }\;\;\max{|z_1-z_2|=K_1K_2+\rho_1+\rho_2} \] 
\item Αν οι κύκλοι εφάπτονται εξωτερικά δηλαδή $ K_1K_2=\rho_1+\rho_2 $ τότε : $ \max{|z_1-z_2|}=2\rho_1+2\rho_2 $
\item Αν οι κύκλοι εφάπτονται εσωτερικά δηλαδή $ Κ_1Κ_2=\rho_1-\rho_2 $ τότε : $ \max{|z_1-z_2|}=2\rho_1 $ όπου $ \rho_1 $ η ακτίνα του μεγαλύτερου κύκλου.
\end{enumerate}
\item Αν οι εικόνες των μιγαδικών $ z_1,z_2 $ κινούνται στην έλλειψη $ \dfrac{x^2}{a^2}+\dfrac{y^2}{\beta^2}=1 $ τότε : \[ \max{|z_1-z_2|}=2a\;\;\textrm{ και }\;\;\min{|z_1-z_2|}=2\beta\;\textrm{ αν οι εικόνες των }z_1,z_2\textrm{ είναι αντιδιαμετρικές} \]
\end{enumerate}
\end{enumerate}

\textbf{{\large ΟΡΙΑ - ΣΥΝΕΧΕΙΑ}}
\end{center}
\begin{enumerate}[itemsep=0mm]
\item Πραγματική συνάρτηση : $ f:A\rightarrow\mathbb{R} $ ονομάζεται η διαδικασία με την οποία κάθε τιμή μιας πραγματικής μεταβλητής $ x\in A $ αντιστοιχεί μόνο μια τιμή $ y\in f(A) $.
\begin{enumerate}[itemsep=0mm,label=\roman*.]
\item $ A=D_f $ : πεδίο ορισμού.
\item $ f(D_f) $ : σύνολο τιμών : $ f(D_f)=\left\lbrace y\in\mathbb{R}|y=f(x)\;,\;x\in D_f\right\rbrace  $
\item $ y=f(x) $ : τιμή της $ f $ στο $ x $.
\item Οι ρίζες της εξίσωσης $ f(x)=0 $ ονομάζονται και ρίζες της συνάρτησης $ f $.
\item Αν $ \kappa\in f(D_f)$ τότε υπάρχει $ x_0\in D_f $ ώστε $ f(x_0)=\kappa $.
\end{enumerate}
\item Είδη συναρτήσεων
\begin{enumerate}[itemsep=0mm,label=\roman*.]
\item Πολυωνυμική : $ f(x)=a_\nu x^\nu+a_{\nu-1}x^{\nu-1}+\ldots+a_1x+a_0 $.
\item Ρητή : $ f(x)=\dfrac{g(x)}{h(x)} $ με $ h(x)\neq0 $.
\item Άρρητη : $ f(x)=\!\sqrt{g(x)} $ με $ g(x)\geq0 $.
\item Λογαριθμική : $ f(x)=\ln{g(x)} $ με $ g(x)>0 $.
\item Εκθετική : $ f(x)=g(x)^{h(x)} $ με $ g(x)>0 $.
\item Τριγωνομετρική : $ f(x)=\hm{g(x)}\;\textrm{ ή }\;\syn{g(x)}\;\textrm{ ή }\;\ef{g(x)}\;\textrm{ ή }\;\syf{g(x)} $.
\item Σταθερή : $ f(x)=c $ με $ c\in\mathbb{R} $.
\item Μηδενική : $ f(x)=0 $.
\item Ταυτοτική : $ f(x)=x $ για κάθε $ x\in\mathbb{R} $.
\end{enumerate}
\item Γραφική παράσταση συνάρτησης : Σύνολο σημείων $ Μ\left( x,f(x)\right)\;,\;C_f=\{M\left( x,y\right)/y=f(x)\}$. Ισχύει \[ Μ(x_0,y_0)\in C_f\Leftrightarrow y_0=f(x_0) \]
\item Άρτια - Περιττή συνάρτηση :\\
Αν $ \forall x\in D_f\Rightarrow -x\in D_f $ και
\begin{enumerate}[itemsep=0mm,label=\roman*.]
\item $ f(-x)=f(x) $ η $ f $ είναι άρτια. Είναι συμμετρική ως προς $ y'y $.
\item $ f(-x)=-f(x) $ η $ f $ είναι περιττή. Είναι συμμετρική ως προς το $ O $.
\end{enumerate}
\item Ίσες συναρτήσεις. $ f,g $ ίσες όταν
\begin{enumerate}[itemsep=0mm,label=\roman*.]
\item $ D_f=D_g $ και 
\item $ \forall x\in D_f  \Rightarrow f(x)=g(x)$.
\end{enumerate}
\item Πράξεις με συναρτήσεις : Αν $ f:D_f\rightarrow\mathbb{R} $ και $ g:D_g\rightarrow \mathbb{R} $ τότε :
\begin{enumerate}[itemsep=0mm,label=\roman*.]
\item $ (f+g)(x)=f(x)+g(x)\;\;,\;\;D_{f+g}=D_f\cap D_g $.
\item $ (f-g)(x)=f(x)-g(x)\;\;,\;\;D_{f-g}=D_f\cap D_g $.
\item $ (f\cdot g)(x)=f(x)\cdot g(x)\;\;,\;\;D_{f+g}=D_f\cap D_g $.
\item $ \left( \dfrac{f}{g}\right) (x)=\dfrac{f(x)}{g(x)}\;\;,\;\;D_{f/g}=\{x\in D_f\cap D_g/g(x)\neq0\} $.
\end{enumerate}
\item Σύνθεση συναρτήσεων : Αν $ f:D_f\rightarrow\mathbb{R} $ και $ g:D_g\rightarrow \mathbb{R} $ τότε :\\
Η σύνθεση $ f\circ g $ είναι :
\begin{enumerate}[itemsep=0mm,label=\roman*.]
\item $ D_{f\circ g}=\{x\in D_g/g(x)\in D_f\} $
\item $ (f\circ g)(x)=f\left( g(x)\right)  $
\end{enumerate}
Η σύνθεση $g\circ f $ είναι :
\begin{enumerate}[itemsep=0mm,label=\roman*.]
\item $ D_{g\circ f}=\{x\in D_f/f(x)\in D_g\} $
\item $ (g\circ f)(x)=g\left( f(x)\right)  $
\end{enumerate}
Ισχύει επίσης $ (f\circ g)\circ h=f\circ (g\circ h) $.
\item Μονοτονία συνάρτησης
\begin{enumerate}[itemsep=0mm,label=\roman*.]
\item $ f:\varDelta\rightarrow\mathbb{R} $ ($ \varDelta $ διάστημα) αύξουσα αν $ \forall x_1,x_2\in\varDelta $ ισχύει $ x_1<x_2\Rightarrow f(x_1)<f(x_2) $.
\item $ f:\varDelta\rightarrow\mathbb{R} $ ($ \varDelta $ διάστημα) φθίνουσα αν $ \forall x_1,x_2\in\varDelta $ ισχύει $ x_1<x_2\Rightarrow f(x_1)>f(x_2) $.
\end{enumerate}
\item Μονοτονία - Λύση εξισώσεων, ανισώσεων.\\
Αν $ f:\varDelta\rightarrow\mathbb{R} $ μια γνησίως μονότονη συνάρτηση τότε :
\begin{enumerate}[itemsep=0mm,label=\roman*.]
\item Αν $ f\uparrow\mathbb{\varDelta} $ τότε $ f(x)<f(x_0)\Leftrightarrow x<x_0 $.
\item Αν $ f\downarrow\mathbb{\varDelta} $ τότε $ f(x)<f(x_0)\Leftrightarrow x>x_0 $.
\item Η εξίσωση $ f(x)=0 $ έχει μια το πολύ λύση στο διάστημα $ \varDelta $.
\item Αν $ 0\in f(\varDelta) $ τότε η εξίσωση $ f(x)=0 $ έχει μια ακριβώς ρίζα στο $ \varDelta $.
\item Αν $ \eta\in f(\varDelta) $ τότε η εξίσωση $ f(x)=\eta $ έχει μια ακριβώς ρίζα στο $ \varDelta $.
\end{enumerate}
\item Ακρότατα συνάρτησης - Μέγιστο, ελάχιστο.\\
Για μια συνάρτηση $ f:\varDelta\rightarrow\mathbb{R} $ έχουμε :
\begin{enumerate}[itemsep=0mm,label=\roman*.]
\item $ f(x_0) $ μέγιστο όταν $ f(x)\leq f(x_0)\;,\;\forall x\in\varDelta$.
\item $ f(x_0) $ ελάχιστο όταν $ f(x)\geq f(x_0)\;,\;\forall x\in\varDelta$.
\end{enumerate}
Αν $ m\leq f(x)\leq M $ και $ m,M\in f(\varDelta) $ τότε $ m $ ελάχιστο και $ M $ μέγιστο.
\begin{enumerate}[itemsep=0mm,label=\roman*.]
\item Αν $ f\uparrow \varDelta=[a,\beta] \Rightarrow \min f=f(a) $ και $ \max f=f(\beta) $.
\item Αν $ f\downarrow \varDelta=[a,\beta] \Rightarrow \min f=f(\beta) $ και $ \max f=f(a) $.
\item Αν $ f $ μονότονη στο $ \varDelta=(a,\beta) $ δεν έχει ακρότατα.
\end{enumerate}
\item Συνάρτηση $ 1-1 $.
\begin{enumerate}[itemsep=0mm,label=\roman*.]
\item $ f:D_f\rightarrow\mathbb{R} \;\; 1-1$ όταν $ \forall x_1,x_2\in D_f $ ισχύει $ x_1\neq x_2\Rightarrow f(x_1)\neq f(x_2) $.
\item $ f:D_f\rightarrow\mathbb{R} \;\; 1-1$ όταν $ \forall x_1,x_2\in D_f $ ισχύει $ f(x_1)=f(x_2)\Rightarrow x_1=x_2 $.
\end{enumerate}
Αν $ f $ μονότονη τότε είναι και $ 1-1 $.
\item Αντίστροφη συνάρτηση : Αν  $ f:D_f\rightarrow\mathbb{R} $ μια συνάρτηση $ 1-1 $ τότε $ f^{-1}:f(D_f)\rightarrow\mathbb{R} $ είναι η αντίστροφή της. 
\begin{enumerate}[itemsep=0mm,label=\roman*.]
\item Η $ f^{-1} $ έχει πεδίο ορισμού το σύνολο τιμών της $ f $ και σύνολο τιμών, το πεδίο ορισμού της $ f $.
\item $ \left( f^{-1}\circ f\right)(x)=x\;\;,\;\;\forall x\in D_f $. 
\item $ \left( f\circ f^{-1}\right)(y)=y\;\;,\;\;\forall y\in f(D_f) $.
\end{enumerate}
\item Όριο συνάρτησης σε σημείο $ x_0 $ : $ \displaystyle{\lim_{x\rightarrow x_0}f(x)=\ccases{\mathcal{l}\in\mathbb{R}\\\pm\infty\\\textrm{δεν υπάρχει}}} $\\

\item Ιδιότητες ορίων

\item Αν $ \displaystyle{\lim_{x\rightarrow x_0}f(x)}<0 $ τότε $ f(x_0)<0 $ κοντά στο $ x_0 $.

\item Πρόσημο συνάρτησης\\
Αν για μια συνάρτηση $ f $ ορισμένη σε ένα διάστημα $ \varDelta $ ισχύει $ f(x)\neq0\;,\;\forall x\in\varDelta $ τότε η $ f $ διατηρεί το πρόσημό της στο $ \varDelta $ δηλαδή :
\[ f(x)>0\textrm{ ή }f(x)<0\;,\;\forall x\in\varDelta \]
\begin{enumerate}[itemsep=0mm,label=\roman*.]
\item Αν υπάρχει $ a\in\varDelta $ ώστε $ f(a)>0 $ τότε $ f(x)>0 $. Αντίστοιχα αν υπάρχει $ a\in\varDelta $ ώστε $ f(a)<0 $ τότε $ f(x)<0 $.\\
\item Αν $ \rho_1,\rho_2 $ είναι δύο διαδοχικές ρίζες της $ f $ τότε η συνάρτηση διατηρεί το πρόσημό της στο διάστημα $ (\rho_1,\rho_2) $ μεταξύ των ριζών.
\end{enumerate}
\item Θεώρημα ενδιάμεσων τιμών\\
Αν μια συνάρτηση $ f $ ορισμένη σε ένα κλειστό διάστημα $ [a,\beta] $ είναι 
\begin{enumerate}[itemsep=0mm,label=\roman*.]
\item συνεχής στο $ [a,\beta] $ και ισχύει
\item $ f(a)\neq f(\beta) $
\end{enumerate}
τότε για κάθε τιμή $ \eta $ μεταξύ των $ f(a) $ και $ f(\beta) $ υπάρχει ένας τουλάχιστον αριθμός $ x_0\in(a,\beta) $ ώστε $ f(x_0)=\eta $.
\begin{enumerate}[itemsep=0mm,label=\roman*.]
\item Η εικόνα $ f(\varDelta) $ ενός διαστήματος $ \varDelta $ μέσω μιας συνεχούς και μη σταθερής συνάρτησης $ f $ είναι διάστημα.
\item Αν μια $ 1-1 $ συνάρτηση $ f $ είναι συνεχής σε ένα διάστημα $ \varDelta $ τότε είναι γνησίως μονότονη.
\end{enumerate}
\item Θεώρημα μέγιστης και ελάχιστης τιμής.\\

\begin{enumerate}[itemsep=0mm,label=\roman*.]
\item Αν $ m=M $ τότε η συνάρτηση είναι σταθερή.
\item Αν $ m\leq f(\eta)\leq M $ τότε $ \eta\in f(\varDelta) $ και υπάρχει $ x_0\in[a,\beta] $ ώστε $ f(x_0)=\eta $.
\item Αν $ m\leq f(\eta)\leq M $ και η $ f $ είναι γνησίως μονότονη στο $ [a,\beta] $ τότε η εξίσωση $ f(x)=\eta $ έχει ακριβώς μια λύση.
\end{enumerate}
\item Σύνολο τιμών - Εικόνα κλειστού - ανοιχτού διαστήματος.
\begin{enumerate}[itemsep=0mm,label=\roman*.]
\item Αν $ \varDelta=[a,\beta] $ τότε
\begin{itemize}
\item Αν $ f\uparrow\varDelta $ τότε $ f(\varDelta)=\left[ f(a),f(\beta)\right] $.
\item Αν $ f\downarrow\varDelta $ τότε $ f(\varDelta)=\left[ f(\beta),f(a)\right]  $.
\end{itemize}
\item Αν $ \varDelta=(a,\beta) $ τότε
\begin{itemize}
\item Αν $ f\uparrow\varDelta $ τότε $ f(\varDelta)=\displaystyle{\left( \lim_{x\rightarrow a^+}f(x),\lim_{x\rightarrow \beta^-}f(x)\right)} $.
\item Αν $ f\downarrow\varDelta $ τότε $ f(\varDelta)=\displaystyle{\left( \lim_{x\rightarrow \beta^-}f(x),\lim_{x\rightarrow a^+}f(x)\right)} $.
\end{itemize}
\end{enumerate}
\end{enumerate}

\item Ορισμένο ολοκλήρωμα : $ \displaystyle{\int_{a}^{\beta}{\!\!\!f(x)\mathrm{d}x}} $ με $ f $ συνεχή στο $ [a,\beta] $.
\begin{multicols}{2}
\begin{enumerate}[itemsep=0mm,label=\roman*.]
\item $ \displaystyle{\int_{a}^{a}{\!\!\!f(x)\mathrm{d}x}=0} $
\item $ \displaystyle{\int_{a}^{\beta}{\!\!\!f(x)\mathrm{d}x}}=-\displaystyle{\int_{\beta}^{a}{\!\!\!f(x)\mathrm{d}x}} $
\item $ \displaystyle{\int_{a}^{\beta}{\!\!\!f(x)\mathrm{d}x}}=\displaystyle{\int_{a}^{\gamma}{\!\!\!f(x)\mathrm{d}x}}+\displaystyle{\int_{\gamma}^{\beta}{\!\!\!f(x)\mathrm{d}x}} $
\end{enumerate}
\end{multicols}
\item Εμβαδόν χωρίου : Αν για την $ f:[a,\beta]\rightarrow\mathbb{R} $ ισχύει $ f(x)\geq0 $ τότε $ \displaystyle{Ε(\varOmega)=\int_{a}^{\beta}{\!\!\!f(x)\mathrm{d}x}} $.

\item Θεμελιώδες θεώρημα ολοκληρωτικού λογισμού\\
Έστω $ f:[a,\beta]\rightarrow\mathbb{R} $ μια συνεχής συνάρτηση στο $ [a,\beta] $. Αν $ G $ είναι μια αρχική της $ f $ στο $ [a,\beta] $ τότε \[ \int_{a}^{\beta}{\!\!\!f(x)\mathrm{d}x}=G(\beta)-G(a) \]
Ισχύει οτι $ \displaystyle{\int_{a}^{x}{\!\!\!f'(x)\mathrm{d}x}=[f(x)]_{a}^{x}\Rightarrow f(x)=\int_{a}^{x}{\!\!\!f'(x)\mathrm{d}x}+f(a)} $.
\item Αν μια συνάρτηση $ f:[a,\beta]\rightarrow\mathbb{R} $ είναι συνεχής στο $ [a,\beta] $ τότε υπάρχει $ \xi\in(a,\beta) $ ώστε \[ \int_{a}^{\beta}{\!\!\!f(x)\mathrm{d}x}=(\beta-a)f(\xi) \]
\item Μέθοδοι ολοκλήρωσης

\end{enumerate}

\item Η συνάρτηση $ F(x)=\displaystyle{\int_{a}^{x}{\!\!\!f(t)\mathrm{d}t}} $\\
Αν $ f:\varDelta\rightarrow\mathbb{R} $ συνεχής στο $ \varDelta $ και $ a $ ένα σημείο του $ \varDelta $ τότε η συνάρτηση \[ F(x)=\displaystyle{\int_{a}^{x}{\!\!\!f(t)\mathrm{d}t}} \] είναι μια αρχική της $ f $ στο $ \varDelta $. Για να ορίζεται η $ F $ πρέπει $ a,x\in\varDelta $.
\begin{enumerate}[itemsep=0mm,label=\roman*.]
\item $ \displaystyle{\left( \int_{a}^{x}{\!\!\!f(t)\mathrm{d}t}\right)'=f(x)} $
\item Αν $ G(x)=\displaystyle{ \int_{a}^{g(x)}{\!\!\!f(t)\mathrm{d}t}} $ τότε $$  \displaystyle{\left( \int_{a}^{g(x)}{\!\!\!f(t)\mathrm{d}t}\right)'=f(g(x))\cdot g'(x)}$$ με $D_G=\{x\in D_g\textrm{ και }g(x)\in\varDelta\} $.
\item Αν $ Η(x)=\displaystyle{ \int_{g_1(x)}^{g_2(x)}{\!\!\!f(t)\mathrm{d}t}} $ τότε $$  \displaystyle{\left( \int_{g_1(x)}^{g_2(x)}{\!\!\!f(t)\mathrm{d}t}\right)'=f(g_2(x))\cdot g'_2(x)-f(g_1(x))\cdot g'_1(x)}$$ με $D_Η=\{x\in D_{g_{1}}\cap D_{g_2}\textrm{ και }g_1(x),g_2(x)\in\varDelta\} $.
\end{enumerate}
