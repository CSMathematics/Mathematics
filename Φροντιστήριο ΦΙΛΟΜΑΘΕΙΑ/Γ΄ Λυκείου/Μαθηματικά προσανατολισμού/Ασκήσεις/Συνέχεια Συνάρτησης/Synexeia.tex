\paragraph{\large ΘΕΩΡΙΑ}
\begin{enumerate}[itemsep=5mm,,leftmargin=0mm,itemindent=5mm]
\item \textbf{Θεωρία}\\
Να απαντήσετε στις παρακάτω ερωτήσεις.
\begin{rlist} 
\item Πότε μια συνάρτηση $ f $ ονομάζεται συνεχής σε ένα σημείο $ x_0 $ του πεδίου ορισμού της?
\item Πότε μια συνάρτηση $ f $ λέγεται συνεχή σε ένα ανοιχτό διάστημα $ (a,\beta) $ του πεδίου ορισμού της?
\item Να διατυπώσετε τον ορισμό της συνέχειας μιας συνάρτησης $ f $ σε ένα κλειστό διάστημα $ [a,\beta] $.
\end{rlist}
\item 
\end{enumerate}
\paragraph{\large ΑΣΚΗΣΕΙΣ}
\begin{enumerate}[itemsep=5mm,leftmargin=0mm,itemindent=5mm]
\item \textbf{Συνέχεια σε σημείο}\\
Έστω μια συνάρτηση $ f:\mathbb{R}\rightarrow\mathbb{R} $ η οποία ορίζεται από τον τύπο
\[ f(x)=\ccases{\dfrac{\sqrt{x^2+3x}-2}{x-1} &\textrm{, αν } x<1\\[3mm]
\ln{xe^x}+\dfrac{1}{3}\hm{\dfrac{\pi x}{6}} &\textrm{, αν } x\geq1} \]
Να εξεταστεί αν η συνάρτηση $ f $ είναι συνεχής στο σημείο $ x_0=1 $.
\item \textbf{Εύρεση παραμέτρου}\\
Δίνεται η συνάρτηση $ f:(1,+\infty)\rightarrow\mathbb{R} $ με τύπο \[ f(x)=\ccases{ax^2+(1-a)\syn{\pi x}-2 & \textrm{, αν }x<2\\
\sqrt{ax-1}+e^{x-2} &\textrm{, αν }x\geq2}\]
Να βρεθεί η τιμή της παραμέτρου $ a $ ώστε η συνάρτηση να είναι συνεχής στο πεδίο ορισμού της.
\item \textbf{Εύρεση παραμέτρου}\\
Έστω μια συνάρτηση $ f $ με πεδίο ορισμού το $ \mathbb{R} $ η οποία ορίζεται από τον τύπο
\[ f(x)=\ccases{(1-a)x^2-\beta\ln{ex^2}  &\textrm{, }x<1\\
3\syn{(2\pi x)}-|x-3|+1 &\textrm{, }1\leq x\leq2\\[1mm]
\dfrac{\sqrt{2x+5}-3}{|x-2|}+ax-\beta &\textrm{, }x>2} \]
Να βρεθούν οι τιμές των παραμέτρων $ a,\beta $ ώστε η συνάρτηση $ f $ να είναι συνεχής.
\item \textbf{Εύρεση τύπου}\\
Δίνεται η συνεχής συνάρτηση $ f:(0,+\infty)\rightarrow\mathbb{R} $ για την οποία ισχύει η σχέση \[ x^2f(x)=|x^2-3x|-2x+f(x) \]
Να βρεθεί ο τύπος της συνάρτησης $ f $ για κάθε $ x\in(0,+\infty) $.
\item \textbf{Αποδεικτική}\\
Έστω μια συνάρτηση $ f $ με πεδίο ορισμού το διάστημα $ [0,+\infty) $ για την οποία ισχύει η σχέση
\[ x^2-x\hm{x}+1\leq x^2f(x)+1\leq \syn^2{x}-2x\syn{x}+2 \]
για κάθε $ x\in[0,+\infty) $. Να δειχθεί οτι η συνάρτηση είναι συνεχής στο σημείο $ x_0=0 $.
\end{enumerate}
