\paragraph{\textcolor{black}{{\large ΑΣΚΗΣΕΙΣ}}}\mbox{}\\
\begin{enumerate}[itemsep=5mm]
\item Να λυθούν οι παρακάτω εξισώσεις
\results{$ z^2-2\overline{z}+1=0 $,$ z^3+z^2-2=0 $,$ z^4=1 $, $ |z+i|=iz $}
\item Να βρεθεί ο γεωμετρικός τόπος των εικόνων των μιγαδικών $ z $ για 
τους οποίους ισχύει : 
\[ Re\PARENS{z+\frac{1}{z}}=5Re(z) \]
\item Να βρεθεί ο γεωμετρικός τόπος τον 
εικόνων των μιγαδικών $ z $ για τους οποίους ισχύει : \[ |z+1-2i|=3 \]
\item Να βρεθεί ο γεωμετρικός τόπος τον 
εικόνων των μιγαδικών $ z $ για τους οποίους ισχύει: \[ |z-2i|=|z+1| \]
\item Έστω Μ και Ν οι εικόνες των μιγαδικών $ z $ και $ w $, για τους οποίους 
ισχύει \[ w=iz-\frac{i}{z} \]
και για το Μ ισχύει $ |z| =1 $ (δηλαδή κινείται σε 
μοναδιαίο κύκλο), να δεχθεί ότι το σημείο Ν κινείται σε ευθύγραμμο τμήμα.
\item Δίνεται ο μιγαδικός αριθμός $ z $ με την ιδιότητα \[ |z-2-2i|=3\!\sqrt{2} \] 
α) Να βρεθεί ο γεωμετρικός τόπος της εικόνας Μ του $ z $.\\
β) Ποίος απ' τους παραπάνω μιγαδικούς έχει το μεγαλύτερο και ποίος το μικρότερο μέτρο?
\item Έστω $ z, w $ μιγαδικοί αριθμοί για τους οποίους ισχύει: 
\[ |(1-i)z-2|=2\;\textrm{ και }\;|w+2i|=|w-2+4i| \] 
α) Να βρεθεί ο γεωμετρικός τόπος των εικόνων των μιγαδικών $ z $.\\ 
β) Να βρεθεί ο γεωμετρικός τόπος των εικόνων των μιγαδικών $ w $.\\
γ) Να βρεθεί η ελάχιστη τιμή του $ |z-w| $.
\end{enumerate}
