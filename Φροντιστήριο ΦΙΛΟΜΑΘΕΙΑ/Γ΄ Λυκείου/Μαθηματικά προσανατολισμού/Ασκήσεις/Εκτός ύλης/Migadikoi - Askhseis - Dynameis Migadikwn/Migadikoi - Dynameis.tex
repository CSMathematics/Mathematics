\item $ i^5 $ \item $ i^9 $ \item $ i^{17} $ \item $ i^{250} $ \item $ i^{1000} $ \item $ i^{2013} $ \item $ i^{-2} $ \item $ i^{-30} $
\end{enumerate}
\end{multicols}
\item Να υπολογιστεί η τιμή των παραστάσεων
\begin{multicols}{2}
\begin{enumerate}[label=\roman*.]
\item $ S=1+i+i^2+\ldots+i^{100} $ \item $ A=i^{2010}+i^{2011}+i^{2012}+i^{2013} $ 
\end{enumerate}
\end{multicols}
\item Να γίνουν οι παρακάτω πράξεις
\begin{multicols}{3}
\begin{enumerate}[label=\roman*.]
\item $ (1-i)^2 $ \item $ (3-2i)^2 $ \item $ (2+i)^3 $ \item $ (1+i\sqrt{3})^3 $ \item $ \PARENS{\dfrac{1}{2}+i\dfrac{\sqrt{3}}{2}}^{100} $ \item $ (1+3i)^6 $ \item $ (1+i)^5 $ \item $ (3-4i)^8 $ \item $ (\sqrt{3}+i)^{10} $
\end{enumerate}
\end{multicols}
\item Να γίνουν οι παρακάτω πράξεις
\begin{multicols}{3}
\begin{enumerate}[label=\roman*.]
\item $ \PARENS{\dfrac{4-i}{1+4i}}^2 $
\item $ \PARENS{\dfrac{1-i\sqrt{3}}{\sqrt{3}+i}}^3 $
\item $ \PARENS{\dfrac{1+2i}{1-2i}}^6 $
\item $ \PARENS{\dfrac{2+5i}{2-5i}}^{2\nu} $
\item $ \PARENS{\dfrac{a-i}{1+ai}}^{20} $
\item $ \PARENS{\dfrac{1-i}{1+i}}^\nu\cdot\PARENS{\dfrac{2-i}{1+2i}}^{-\nu} $
\end{enumerate}
\end{multicols}
\item Να βρεθούν οι τιμές του ακέραιου αριθμού $ \nu\in\mathbb{Z} $ ώστε να ισχύουν οι σχέσεις
\begin{multicols}{3}
\begin{enumerate}[label=\roman*.]
\item $ 2\nu+3=i^2 $
\item $ 4\nu-5=i^4 $
\item $ \nu+1=i^k $
\item $ \nu^2-\nu+1=i^k $
\item $ \nu i+2i=i^3 $
\item $ i(2\nu^2-3\nu)=i^k $
\end{enumerate}
\end{multicols}
\item Να βρεθεί η τιμή της παρακάτω παράστασης
\[ S=1+\PARENS{\dfrac{1}{2}+i\dfrac{\sqrt{3}}{2}}+\PARENS{\dfrac{1}{2}+i\dfrac{\sqrt{3}}{2}}^2+\ldots+\PARENS{\dfrac{1}{2}+i\dfrac{\sqrt{3}}{2}}^{\nu-1} \]
\item Αν $ z=\lambda\cdot\dfrac{1+\lambda i}{\lambda-i} $ να βρεθεί η τιμή του πραγματικού αριθμού $ \lambda\in\mathbb{R} $ ώστε
\[ iz^3-1=0 \]


\end{enumerate}
