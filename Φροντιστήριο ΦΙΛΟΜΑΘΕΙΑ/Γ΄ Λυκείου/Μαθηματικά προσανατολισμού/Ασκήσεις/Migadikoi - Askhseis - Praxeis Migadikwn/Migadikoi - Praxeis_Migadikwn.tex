\begin{enumerate}[label=\bf\textcolor{black}{{\large \arabic*.}},
itemsep=5mm]
\item Να γίνουν οι παρακάτω πράξεις
\begin{multicols}{3}
\begin{enumerate}[label=\roman*.,itemsep=0mm]
\item $ (2-3i)+(4+i) $\\
\item $ (3-2i)+(14-8i) $ \\
\item $ (4-3i)-(2+5i) $ \\
\item $ (17-11i)-(22-9i) $\\
\item $ (3-i)(2+4i) $ \\
\item $ (-4-7i)(2-3i) $\\
\end{enumerate}\end{multicols}
\item Να γίνουν οι παρακάτω πράξεις
\begin{multicols}{2}
\begin{enumerate}[label=\roman*.,itemsep=0mm]
\item $ 2i(2-3i)(1+5i) $\\
\item $ 3i-2(4+7i)-i(4-i) $ \\
\item $ (3-i)(1+4i)(5+3i) $ \\
\item $ -i\left[-(-2+3i)+(4-3i)(1-2i) \right]  $\\
\end{enumerate}\end{multicols}
\item Να γραφτούν οι παρακάτω μιγαδικοί αριθμοί στην απλή μιγαδική μορφή $ a+\beta i $.
\begin{multicols}{4}
\begin{enumerate}[label=\roman*.,itemsep=0mm]
\item $ \dfrac{1-2i}{3+4i} $\\
\item $ \dfrac{2i}{5-12i} $ \\
\item $ \dfrac{1-i}{1+i} $ \\
\item $ \dfrac{3-i}{-2-3i} $\\
\end{enumerate}\end{multicols}
\item Να βρεθεί ο αντίστροφος των μιγαδικών αριθμών
\begin{multicols}{4}
\begin{enumerate}[label=\roman*.,itemsep=0mm]
\item $ 3-4i $\\
\item $ 1+5i $ \\
\item $ 1-i\sqrt{3} $
\item $ 1-xi $\\
\end{enumerate}\end{multicols}
\item Αν $ z,w\in\mathbb{C} $ τρεις μιγαδικοί αριθμοί με $ z=4+i, w=3-2i \textrm{ και } v=2-i $ να βρεθούν οι μιγαδικοί αριθμοί
\begin{multicols}{3}
\begin{enumerate}[label=\roman*.,itemsep=0mm]
\item $ z+2w $\\
\item $ 3z-w+v $ \\
\item $ zw-v $ \\
\item $ 2\overline{z}v-w $\\
\item $ w(z+v) $\\
\item $ \dfrac{1}{z} $\\
\item $ \dfrac{v}{z-w} $\\
\item $ \dfrac{\overline{w}}{1-z} $\\
\item $ w\dfrac{1}{z}+v $\\
\item $ v-\dfrac{1}{z\overline{w}} $\\
\item $ \overline{z}v+wz $\\
\item $ (z+3w)(2v-z) $
\end{enumerate}\end{multicols}
\end{enumerate}
