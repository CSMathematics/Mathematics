\paragraph{\textcolor{black}{{\large ΑΣΚΗΣΕΙΣ}}}\mbox{}\\
\begin{enumerate}[label=\bf\textcolor{black}{{\large \arabic*.}},
itemsep=5mm]
\item Να βρεθούν οι πραγματικοί αριθμοί $ x, y\in\mathbb{R} $ έτσι ώστε να ισχύει
\begin{multicols}{2}
\begin{enumerate}[label=\roman*.,itemsep=0mm]
\item $ (x-3y)+(2x-y)=-1+3i $\\
\item $ (x+2y)+(3x-2y)=7+i $ \\
\item $ (x^2-y)+(x-y)=3+i $ \\
\item $ (x-2yi)(1-2i)+(2-3i)(xi-y)=1+2i $\\
\item $ 2x+i(2x-yi)=3 $\\
\item $ \dfrac{x}{1-2i}+\dfrac{y-1}{3+4i}=1-i $
\end{enumerate}\end{multicols}
\item Να βρεθούν οι πραγματικοί αριθμοί $ x, y\in\mathbb{R} $ έτσι ώστε να ισχύει
\begin{multicols}{2}
\begin{enumerate}[label=\roman*.,itemsep=0mm]
\item $ i(x-2yi)+\dfrac{x+1}{1-2i}-\dfrac{3y+1}{1+i}=2i $\\
\item $ (1-xi)^2+(y-2i)^2=1-8i $
\item $ (x-yi)^{10}(x+yi)^{10}+xi=1+i $\\
\item $ (x+yi)^2=3+4i $
\end{enumerate}\end{multicols}
\item Δίνεται ο μιγαδικός αριθμός \[ z=(k^2-3k+4)+(k^2-16)i \]
Να βρεθεί ο πραγματικός αριθμός $ k\in\mathbb{R} $ έτσι ώστε να ισχυει $ z=0 $.
\item Να βρεθούν οι τιμές της γωνίας $ \theta $ με $ \theta\in\left[ \frac{\pi}{2},\frac{3\pi}{2} \right] $ ώστε \[ \left(\eta\mu\theta-i\sigma\upsilon\nu\theta\right)^2-1=0 \]
\item Δίνονται οι μιγαδικοί αριθμοί \[ z=x+2i\quad,\quad w=1-yi\quad,\quad v=x+(y-1)i \]
Να βρεθούν οι πραγματικοί αριθμοί $ x, y\in\mathbb{R} $ ώστε να ισχύει $ z\overline{w}+2v=0 $
\item Δίνεται ο μιγαδικός αριθμός \[ z=\frac{1-(x-2)i}{1+xi}-1-k \]
Να δειχθεί οτι η εξίσωση $ Re(z)=0 $ έχει 2 πραγματικές λύσεις αν και μόνο αν $ k\in\mathbb{R}-\left\lbrace2\right\rbrace $
\end{enumerate}
