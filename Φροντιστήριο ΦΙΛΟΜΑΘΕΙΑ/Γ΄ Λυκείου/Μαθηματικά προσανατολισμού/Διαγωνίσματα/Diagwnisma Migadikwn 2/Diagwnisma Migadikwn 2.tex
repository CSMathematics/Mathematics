\paragraph{\textcolor{black}{{\large ΘΕΜΑΤΑ}}}\mbox{}\\
\begin{enumerate}[label=\bf\textcolor{black}{{\large \arabic*.}},
itemsep=5mm]
\item Να βρεθούν οι πραγματικοί αριθμοί $ x, y\in\mathbb{R} $ ώστε να ισχύουν οι παρακάτω ισότητες.
\begin{enumerate}[label=\roman*.]
\item $ (3-2i)(2x-yi)+1=2(2x-yi)+2i $
\item $ x(1+xi)+2(1+xi)=2(3y-1)i-y(1+yi) $
\end{enumerate}
\item Να γίνουν oι πράξεις \[ \PARENS{\frac{1-5i}{3-2i}}^2+\PARENS{\frac{4-2i}{1-3i}}^2 \]
\item Να λυθεί η παρακάτω εξίσωση \[ (3-2i)^2z+(1+i)^6+(1-i)z=45-3i \]
\item Δίνεται ο μιγαδικός αριθμός $ z=x+yi $ με $ x, y\in\mathbb{R} $ και ο $ w=z^2+2\overline{z}+1 $. Να αποδείξεις οτι αν $ w\in\mathbb{R} $ τότε $ x=1 $ ή $ y=0 $.
\item Να λυθούν οι παρακάτω εξισώσεις
\begin{multicols}{2}
\begin{enumerate}[label=\roman*.,itemsep=0mm]
\item $ 2z^2-z-1=0 $\\
\item $ z^2=5-12i $ 
\end{enumerate}\end{multicols}
\item Να λύθεί η εξίσωση \[ z^2+2\overline{z}+1=0 \]
\item Να βρεθεί ο γεωμετρικός τόπος των εικόνων των μιγαδικών $ z $ για τους οποίους ισχύει \[ Im\PARENS{z-\frac{1}{z}}=2Re(iz) \]
\item Δίνονται οι μιγαδικοί $ z, w $ με $ w=\dfrac{z+1}{z-2i} $. Να βρεις το γεωμετρικό τόπο των εικόνων του $ z $ ώστε
\begin{multicols}{2}
\begin{enumerate}[label=\roman*.,itemsep=0mm]
\item $ w\in\mathbb{R} $\\
\item $ w\in I $
\end{enumerate}\end{multicols}
\end{enumerate}
