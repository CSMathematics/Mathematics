 \begin{enumerate}[label=\bf\textcolor{royalblue}{{\large B.\arabic*}},
itemsep=3mm,leftmargin=-.7cm]
\item Λύνοντας την πρώτη εξίσωση έχουμε \[ (3x-1)\left(8x^2-6x+1\right)=0\Rightarrow\ccases{3x-1=0\Rightarrow x=\frac{1}{3}\\8x^2-6x-1=0\Rightarrow x=\frac{1}{2}\;\textrm{ ή }\;x=\frac{1}{4}}  \]
επομένως το σύνολο των λύσεων της πρώτης εξίσωσης είναι $ \left\lbrace\frac{1}{3},\frac{1}{2},\frac{1}{4}\right\rbrace $. Επίσης για τα σύνολα $ A,A\cap B $ και $ A\cup B $ ισχύει η σχέση $ A\cap B\subseteq A\subseteq A\cup B $. Επομένως θα ισχύει \[ A\cap B\subseteq A\subseteq A\cup B\Rightarrow P(A\cap B)\leq P(A)\leq P(A\cup B) \] επομένως οι παραπάνω πιθανότητες είναι αντίστοιχα \[ P(A\cap B)=\frac{1}{4}\;\;,\;\;P(A)=\frac{1}{3}\;\;,\;\;P(A\cup B)=\frac{1}{2} \]
\item Για την πιθανότητα του ενδεχομένου $ Α'-Β' $ ισχύει \begin{gather*} P(A'-B')=P(A')-P(A'\cap B')=P(A')-P\left((A\cup B)'\right)=\\
1-P(A)-1+P(A\cup B)=P(A\cup B)-P(A)=\frac{1}{2}-\frac{1}{3}=\frac{1}{6}
\end{gather*}
Επίσης το ενδεχόμενο $ \varDelta : $ να πραγματοποιείται το πολύ ένα από τα δύο ενδεχόμενα $ Α $ και $ Β $ μεταφράζεται συμβολικά $ \varDelta=(A\cap B)' $. Επομένως \[ P(\varDelta)=P\left( (A\cap B)'\right)=1-P(A\cap B)=1-\frac{1}{4}=\frac{3}{4} \]
\item Το ενδεχόμενο $ E : $ να πραγματοποιείται μόνο ένα από τα δύο ενδεχόμενα $ Α $ και $ Β $ μεταφράζεται συμβολικά $ E=(A-B)\cup(B-A) $. Οπότε για την πιθανότητα του $ E $ θα ισχύει 
\begin{gather*} 
	P(E)=P\left((A-B)\cup(B-A)\right)=P(A)-P(A\cap B)+P(B)-P(A\cap B)=\\P(A)+P(B)-2P(A\cap B) 
\end{gather*}
Επίσης από τον προσθετικό νόμο τα έχουμε \[ P(A\cup B)=P(A)+P(B)-P(A\cap B)\Rightarrow \frac{1}{2}=\frac{1}{3}+P(B)-\frac{1}{4}\Rightarrow P(B)=\frac{5}{12} \]
Άρα προκύπτει $$ P(E)=\frac{1}{3}+\frac{5}{12}-2\cdot\frac{1}{4}=\frac{1}{4} $$
\item Οι λύσεις της δεύτερης εξίσωσης είναι \[ 9x^2-3x-2=0\Rightarrow x=\frac{2}{3}\;\textrm{ ή };x=-\frac{1}{3} \]
όμως η δεύτερη λύση απορρίπτεται λόγω του ότι $ 0\leq P(\varGamma)\leq 1 $ άρα \[ x=\frac{2}{3}\Rightarrow P(\varGamma)=\frac{2}{3} \]
Έστω οτι τα ενδεχόμενα $ B $ και $ \varGamma $ είναι ασυμβίβαστα. Τότε από τον προσθετικό νόμο θα ισχύει \[ P(B\cup\varGamma)=P(B)+P(\varGamma)=\frac{5}{12}+\frac{2}{3}=\frac{13}{12}>1 \] άτοπο. Επομένως τα ενδεχόμενα $ B $ και $ \varGamma $ δεν είναι ασυμβίβαστα.
\end{enumerate}
