\begin{enumerate}
\item Μέση τιμή 
\begin{alist}
\item από παρατηρήσεις $ t_i: \ \ \bar{x}=\dfrac{t_1+t_2+\ldots+t_\nu}{\nu} $
\item από συχνότητες $ \nu_i:\ \ \bar{x}=\dfrac{x_1\nu_1+x_2\nu_2+\ldots+x_{\kappa}\nu_{\kappa}}{\nu} $
\item από σχετικές συχνότητες $ f_i:\ \ \bar{x}=x_1f_1+x_2f_2+\ldots+x_{\kappa}f_{\kappa} $
\end{alist}
\item Διάμεσος
\begin{alist}
\item με άρτιο πλήθος $ \nu:\ \ \delta=t_{\frac{\nu+1}{2}} $
\item με περιττό πλήθος $ \nu:\ \ \delta=\dfrac{t_{\frac{\nu}{2}}+t_{\frac{\nu+2}{2}}}{2} $
\end{alist}
\item Σταθμικός μέσος $ \bsr{x}=\dfrac{x_1w_1+x_2w_2+\ldots+x_{\nu}w_{\nu}}{w_1+w_2+\ldots+w_{\nu}} $, όπου $ x_1,x_2,\ldots,x_{\nu} $ είναι οι \textbf{τιμές} και $ w_1,w_2,\ldots,w_{\nu} $ οι συντελεστές βαρύτητάς τους
\item Εύρος $ R=t_{\max}-t_{\min} $
\item Επικρατούσα τιμή : $ M_0= $ η τιμή με τη μεγαλύτερη συχνότητα.
\item Διακύμανση
\begin{alist}
\item από παρατηρήσεις $ t_i:\ \ s^2=\dfrac{(t_1-\bar{x})^2+(t_2-\bar{x})^2+\ldots+(t_{\nu}-\bar{x})^2}{\nu} $
\item από συχνότητες $ \nu_i:\ \ s^2=\dfrac{(x_1-\bar{x})^2\nu_1+(x_2-\bar{x})^2\nu_2+\ldots+(x_{\kappa}-\bar{x})^2\nu_{\kappa}}{\nu} $
\end{alist}
\item Τυπική απόκλιση $ s=\sqrt{s^2} $
\item Συντελεστής μεταβολής $ CV=\dfrac{s}{|\bar{x}|} $
\end{enumerate}
