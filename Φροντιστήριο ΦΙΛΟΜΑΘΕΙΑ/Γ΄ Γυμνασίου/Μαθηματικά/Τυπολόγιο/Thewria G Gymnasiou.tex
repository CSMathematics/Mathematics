\section{Πράξεις με πραγματικούς αριθμούς}
\orismoi
\Orismos{Σύνολα αριθμών}
\vspace{-5mm}
\begin{enumerate}[itemsep=0mm]
\item \textbf{Φυσικοί Αριθμοί} : Το σύνολο των αριθμών από το 0 εως το άπειρο όπου κάθε αριθμός προκύπτει από τον προηγούμενο προσθέτοντας 1 μονάδα. Συμβολίζεται με $ \mathbb{N} $ και είναι : $ \mathbb{N}=\{0,1,2,\ldots\} $.
\item \textbf{Ακέραιοι Αριθμοί} : Το σύνολο των φυσικών αριθμών μαζί με τους αντίθετους τους. Συμβολίζεται με $ \mathbb{Z} $ και είναι : $ \mathbb{Z}=\{\ldots,-1,-1,0,1,2,\ldots\} $.
\item \textbf{Ρητοί Αριθμοί} : Όλοι οι αριθμοί που μπορούν να γραφτούν με τη μορφή κλάσματος με ακέραιους όρους. Συμβολίζεται με $ \mathbb{Q} $ και είναι : $ \mathbb{Q}=\LEFTRIGHT\{\}{\left.\frac{a}{\beta}\right| a,\beta\in\mathbb{Z},\beta\neq0\;} $.
\item \textbf{¶ρρητοι Αριθμοί} : Οποιοσδήποτε αριθμός δεν είναι ρητός. Κατά κύριο λόγο, άρρητοι αριθμοί είναι οι ρίζες που δεν έχουν ακέραιο αποτέλεσμα, ο αριθμός π κ.τ.λ.
\item \textbf{Πραγματικοί Αριθμοί} : Οι ρητοί μαζί με το σύνολο των άρρητων μας δίνουν τους
πραγματικούς αριθμούς, όλους τους αριθμούς που γνωρίζουμε.
\end{enumerate}
\Orismos{Απόλυτη Τιμή Πραγματικού Αριθμού.} Απόλυτη τιμή ενός πραγματικού αριθμού $a$ ορίζεται να είναι η απόσταση του αριθμού αυτού απο το 0 και συμβολίζεται με $\left| a \right| $. \[ \left| a\right| =\ccases{ a & a\geq 0 \\ -a & a<0} \]
Η απόλυτη τιμή ενός πραγματικού αριθμού είναι θετικός αριθμός αφού εξΆ ορισμού παριστάνει απόσταση, που σαν μέγεθος παιρνει μόνο θετικές τιμές.\\\\
\Orismos{Δύναμη πραγματικού αριθμού}Δύναμη με \textbf{βάση} ένα πραγματικό αριθμό $a$ και \textbf{εκθέτη} ένα πραγματικό αριθμό $\nu$ λέγεται το γινόμενο ν παραγόντων ίσων με $ a $ και συμβολίζεται με $ a^\nu $. \[ a^\nu=\undercbrace{a\cdot a\cdot\ldots\cdot a}_{\nu \textrm{ - παράγοντες}}.\]
\Orismos{Τετραγωνική ρίζα}Τετραγωνική ρίζα ενός θετικού αριθμού $x$ ονομάζεται ο θετικός 
αριθμός $ a $ που αν υψωθεί στο τετράγωνο δίνει τον αριθμό $x$ (υπόριζο) και συμβολίζεται με $\sqrt{x}$.
 \[\sqrt{x}=α\qquad\textrm{ όπου }x\geq0\textrm{ και }a\geq0\]

\begin{itemize}[itemsep=0mm]

\item Δύο αριθμοί που έχουν άθροισμα 0 λέγονται \textbf{αντίθετοι}.
\item Δύο αριθμοί που έχουν γινόμενο 1 λέγονται \textbf{αντίστροφοι}.
\end{itemize}\vspace{3mm}

Για κάθε δυναμη με βάση έναν πραγματικό αριθμό $ a $ ορίζουμε\\
$a^1=a \qquad a^0=1\;(\textrm{ με } a\neq 0) \qquad \textrm{ και } \qquad a^{-\nu}=\frac{1}{a^\nu}\;(\textrm{ με } a\neq 0) $\\\\
Επίσης για κάθε δυναμη με εκθέτη ακέραιο αριθμό και εφόσον ορίζεται, ισχύουν οι παρακάτω ιδιότητες για κάθε $ a,\beta\in\mathbb{R} $ και $ \nu,\mu\in\mathbb{Z} $:
\begin{center}
\begin{tabular}{cccc}\hline\rule[-1.5ex]{0pt}{4ex} \textbf{Αρ.} & \textbf{Ιδιότητες} & \textbf{Αρ.} & \textbf{Ιδιότητες} \\ 
\hline\hline\rule[-2ex]{0pt}{7ex}1 & $a^\nu\cdot a^\mu=a^{\nu+\mu}$ & 4 & $ \left( \dfrac{a}{\beta}\right) ^\nu=\dfrac{a^\nu}{\beta^\nu} $\\

\Thewrhma{Ιδιότητεσ Τετραγωνικήσ Ρίζασ}\vspace{-7mm}
 \begin{itemize}[itemsep=0mm]
 \item Δεν ορίζεται τετραγωνική ρίζα αρνητικού αριθμού.
 \item $ \sqrt{x}=a \Leftrightarrow a^2=x $
 \item $ \sqrt{x^2}=\left|x \right|  $
 \item $ \left(\!\sqrt{x}\;\right)^2=x $ για $ x\geq 0 $.
 \end{itemize}
\Thewrhma{τετραγωνικη ριζα γινομενου - πηλικου}
Για κάθε ζευγος θετικών αριθμών  ισχύουν τα παρακάτω
\begin{enumerate}
\item Το γινόμενο των ριζών δυο θετικών αριθμών  είναι ίσο με την τετραγωνική ρίζα του γινομένου τους
\[ \sqrt{a\cdot\beta}=\sqrt{a}\cdot\sqrt{\beta} \]
\item Το πηλίκο των ριζών δύο θετικών πραγματικών αριθμών  είναι ίσο με την τετραγωνική ρίζα του πηλίκου τους. \[ \SQRT{\frac{a}{\beta}}=\frac{\sqrt{a}}{\sqrt{\beta}} \]
\end{enumerate}
Το παραπάνω θεώρημα δεν ισχύει για το άθροισμα ή τη διαφορά των τετραγωνικών ριζών δύο αριθμών δηλαδή έχουμε : \[ \sqrt{a\pm\beta}\neq\sqrt{a}\pm\sqrt{\beta} \]\mbox{}\\
\methodologia
\Methodologia{προσθεση - πολλαπλασιασμοσ}
Για τις πράξεις μεταξύ πραγματικών αριθμών ισχύουν τα παρακάτω :
\begin{center}
\textbf{ΠΡΟΣΘΕΣΗ}
\end{center}
\begin{itemize}[itemsep=0mm]
\item Για να προσθέσουμε δύο ομόσημους αριθμούς, προσθέτουμε τις απόλυτες τιμές τους και τοποθετουμε στο αποτέλεσμα το κοινό τους πρόσημο.
\item Για να προσθέσουμε δύο ετερόσημους αριθμούς, αφαιρούμε τις απόλυτες τιμές τους και βάζουμε στο αποτέλεσμα το πρόσημο του αριθμού με τη μεγαλύτερη απόλυτη τιμή.
\end{itemize}
\begin{center}
\textbf{ΠΟΛΛΑΠΛΑΣΙΑΣΜΟΣ}
\end{center}
\begin{itemize}[itemsep=0mm]
\item Όταν πολλαπλασιάζουμε δύο ομόσημους αριθμούς το αποτέλεσμα είναι θετικό.
\item Όταν πολλαπλασιάζουμε δύο ετερόσημους αριθμούς το αποτέλεσμα είναι αρνητικό.
\end{itemize}
Οι πράξεις της αφαίρεσης και της διαίρεσης προκύπτουν από τις πράξεις της πρόσθεσης και του πολλαπλασιασμού :
\begin{itemize}[itemsep=0mm]
\item Για να προσθέσουμε δύο πραγματικούς αριθμούς προσθέτουμε στον πρώτο τον αντίθετο του δεύτερου.
\[ a-\beta=a+(-\beta) \]
\item Για να διαιρέσουμε δύο πραγματικούς αριθμούς, πολλαπλασιάζουμε τον πρώτο με τον αντίστροφο του δεύτερου.\[ \frac{a}{\beta}=a\cdot\frac{1}{\beta} \]
\end{itemize}
\Methodologia{Δυνάμεισ}Σε αριθμητικές ή αλγεβρικές παραστάσεις οι οποίες περιέχουν δυνάμεις, προσπαθούμε όσο είναι δυνατόν να εκμεταλευτούμε τις ιδιότητες των δυνάμεων, μετασχηματίζοντας τες κατάλληλα ώστε να προκύψουν κοινές βάσεις ή κοινοί εκθέτες.\\\\
\Methodologia{απλοποιηση τετραγωνικήσ ριζασ}
Εαν στο υπόριζο μιας τετραγωνικής ρίζας υπάρχει μεγάλος αριθμός και θέλουμε για υπολογιστικούς ή άλλους λόγους να απλοποιήσουμε τον αριθμό αυτό τότε ακολουθούμε τους παρακάτω τρόπους :
\begin{enumerate}
\item Προσπαθούμε να γράψουμε το υπόριζο σαν γινόμενο δύο παραγόντων, εκ των οποίων ο ένας θα είναι τετράγωνο ακέραιου αριθμού.\\Τότε εκμεταλευόμενοι τη σχέση γινομένου του \textcolor{cyan}{\textbf{Θεωρήματος 1.3.2}} χωρίζουμε τη ρίζα σε γινόμενο ριζών. Έτσι, η τετραγωνική ρίζα με το τετράγωνο του ενός ακεραίου θα απαλοιφθούν με αποτέλεσμα να γραφτεί ο αριθμός σε απλούστερη μορφή.
\begin{enumerate}[itemsep=0mm,label=\roman*.]
\item Για να βρούμε το κατάλληλο ζεύγος αριθμών με τους οποίους θα αντικαταστήσουμε το υπόριζο, θα υπολογίσουμε τους διαιρέτες του σε ζεύγη (διαιρώντας το υπόριζο με ένα διαιρέτη το πηλίκο είναι επίσης διαιρέτης).
\item Όταν με τη μέθοδο αυτή θα συναντήσουμε τετράγωνο ακέραιου αριθμού τότε το ζεύγος στο οποίο ανήκει είναι το ζητούμενο.
\end{enumerate}
Για παράδειγμα να απλοποιήσουμε το $ \sqrt{300} $ έχουμε :
\[ \textrm{Διαιρέτες 300} :  1,2,\textbf{3}\ldots,\textbf{100},150,300  \]
Επομένως η απλοποίηση θα έχει ως εξής: \[ \sqrt{300}=\!\sqrt{3\cdot100}=\sqrt{3}\cdot\!\sqrt{100}=10\!\sqrt{3} \]
\item Ενας άλλος τρόπος να απλοποιήσουμε μια τετραγωνική ρίζα είναι να αναλύσουμε το υπόριζο σε γινόμενο πρώτων παραγόντων. Τις δυνάμεις που θα προκύψουν τις μετατρέπουμε κατάλληλα ώστε να εμφανιστούν ζυγοί εκθέτες, οι οποίοι αν γραφούν σαν πολλαπλάσια του 2 θα απαλοιφθούν με την τετραγωνική ρίζα.\\
Ομοίως για το $ \sqrt{300} $ έχουμε :\[ \sqrt{300}=\!\sqrt{2^2\cdot3\cdot5^2}=\sqrt{2^2}\cdot\!\sqrt{3}\cdot\!\sqrt{5^2}=2\cdot\!\sqrt{3}\cdot\!\sqrt{5}=10\!\sqrt{3} \]
\end{enumerate}
\Methodologia{Ρητοποίηση Παρονομαστή}Για να μετατρέψουμε ένα κλάσμα με άρρητο παρονομαστή της μορφής $ \sqrt{a} $ σε ρητό παρονομαστή, πολλαπλασιάζουμε και τους δύο όρους του κλάσματος με τη ρίζα που βρίσκεται στον παρονομαστή, καταφέρνοντας έτσι να υψωθεί ο παρονομαστής στο τετράγωνο οπότε απαλοίφοντας ρίζα και δύναμη  μενει στον παρονομαστή το ρητό υπόριζο.
\[ \frac{1}{\sqrt{a}}=\frac{\sqrt{a}\cdot1}{\sqrt{a}\cdot\!\sqrt{a}}=\frac{\sqrt{a}}{\left(\!\sqrt{a}\;\right)^2}=\dfrac{\sqrt{a}}{a} \]
\section{Μονώνυμα}
\orismoi
\Orismos{Αλγεβρικέσ Παραστάσεισ.}
\vspace{-7mm}
\begin{enumerate}[itemsep=0mm]
\item \textbf{Αλγεβρικές παραστάσεις} ονομάζονται οι παραστάσεις που περιέχουν αριθμούς και μεταβλητές με πράξεις μεταξύ τους.
\item \textbf{Ακεραία αλγεβρική παράσταση} καλείται μια αλγεβρική παράσταση η οποία έχει μόνες πράξεις τον πολλαπλασιασμό και την πρόσθεση ανάμεσα στις μεταβλητές και με εκθέτες φυσικούς αριθμούς.
\end{enumerate}
\Orismos{Μονώνυμο.} Μονώνυμο ονομάζεται κάθε ακέραια αλγεβρική παράσταση που ανάμεσα στις μεταβλητές έχει \textbf{μόνο} την πράξη του πολλαπλασιασμού.
\begin{itemize}[itemsep=0mm]
\item \textbf{Κύριο μέρος} ενός μονωνύμου ονομάζεται το σύνολο των μεταβλητών του.
\item \textbf{Συντελεστής} ενός μονωνύμου ονομάζεται ο σταθερός αριθμός με τον οποίο πολλαπλασιάζουμε το κύριο μέρος ενός μονωνύμου.
\item Βαθμός ενός μονωνύμου \textbf{ως προς κάποια μεταβλητή} του ονομάζεται ο εκθέτης της μεταβλητής αυτής.
\item Βαθμός ενός μονωνύμου \textbf{ως προς όλες τις μεταβλητές του} ονομάζεται το άθροισμα των βαθμών κάθε μεταβλητής.
\item Οι πραγματικοί αριθμοί ονομάζονται και \textbf{σταθερά} μονώνυμα και είναι μηδενικού βαθμού, ενώ το $ 0 $ ονομάζεται \textbf{μηδενικό} μονώνυμο και δεν έχει βαθμό.
\end{itemize}
\Orismos{Όμοια - Ίσα - Αντίθετα Μονώνυμα.}
\vspace{-7mm}
\begin{itemize}[itemsep=0mm]
\item \textbf{Όμοια} ονομάζονται τα μονώνυμα που έχουν το ίδιο κύριο μέρος. \[ ax^\nu y^\mu\;,\; \beta x^\nu y^\mu \]
\item \textbf{Ίσα} ονομάζονται δύο ή περισσότερα \textbf{όμοια} μονώνυμα που έχουν ίσους συντελεστές.
\item \textbf{Αντίθετα} ονομάζονται δύο ή περισσότερα \textbf{όμοια} μονώνυμα που έχουν αντίθετους συντελεστές.
\end{itemize}
\methodologia
\Methodologia{Πρόσθεση - Αφαίρεση μονωνύμων}
Εαν δύο ή περισσότερα μονώνυμα είναι όμοια, τα προσθέσουμε ή  τα αφαιρέσουμε, κάνοντας πράξεις μόνο με τους συντελεστές τους. Το αποτέλεσμα θα είναι μονώνυμο, όμοιο με τους όρους του αθροίσματος και θα έχει συντελεστή το άθροισμα των συντελεστών τους.
\[ ax^\nu y^\mu\pm\beta x^\nu y^\mu=(a\pm\beta)x^\nu y^\mu \]
\Methodologia{Πολλαπλασιασμόσ - Διαίρεση μονωνύμων}Στον πολλαπλασιασμό και στη διαίρεση μονωνύμων εκτελούμε τις πράξεις κάνοντας χρήση των ιδιοτήτων των δυνάμεων $ a^\nu\cdot a^\mu=a^{\nu+\mu} $ και $ a^\nu\cdot a^\mu=a^{\nu+\mu} $ αντίστοιχα, μεταξύ των αριθμών και μεταξύ των μεταβλητών, χωρίς απαραίτητη συνθήκη την ομοιότητα των μονωνύμων.
\begin{itemize}[itemsep=0mm]
\item Το γινόμενο μονωνύμων έιναι μονώνυμο.
\item Το πηλίκο μονωνύμων δεν είναι πάντοτε μονώνυμο.
\end{itemize}
\section{Πολυώνυμα}
\orismoi
\Orismos{Πολυώνυμο} Πολυώνυμο ονομάζεται η ακέραια αλγεβρική παράσταση η οποία είναι άθροισμα \textbf{ανόμοιων} μονωνύμων.
\begin{itemize}
\item Κάθε μονώνυμο μέσα σ' ένα πολυώνυμο ονομάζεται \textbf{όρος} του πολυωνύμου.
\item Το πολυώνυμο με 3 όρους ονομάζεται \textbf{τριώνυμο}.
\item Οι αριθμοί ονομάζονται \textbf{σταθερά} πολυώνυμα ενώ το 0 \textbf{μηδενικό} πολυώνυμο.
\item Τα πολυώνυμα τα συμβολίζουμε με ένα κεφαλαίο γράμμα όπως :  $ P(x), Q(x), A(x), B(x) $  κτλ. και δίπλα σε μια παρένθεση γράφουμε τη μεταβλητή τους.
\item \textbf{Βαθμός} ενός πολυωνύμου είναι ο βαθμός του μεγιστοβάθμιου όρου.
\item Τα πολυώνυμα μιας μεταβλητής τα γράφουμε κατά φθίνουσες δυνάμεις της μεταβλητής δηλαδή από τη μεγαλύτερη στη μικρότερη.
\[ P(x)=a_\nu x^\nu+a_{\nu-1}x^{\nu-1}+\ldots+a_1x+a_0 \]
\item \textbf{Τιμή} ενός πολυωνύμου ονομάζεται το αποτέλεσμα που προκύπτει ύστερα από πράξεις εαν αντικαταστήσουμε τη μεταβλητή ενός πολυωνύμου με ένα γνωστό αριθμό. Εαν $ x_0 $ ένας σταθερός αριθμός τότε η τιμή του πολυωνύμου $ P(x) $ είναι : \[ P(x_0)=a_\nu x_0^\nu+a_{\nu-1}x_0^{\nu-1}+\ldots+a_1x_0+a_0=\kappa \]
\end{itemize}
\Orismos{Αναγωγή ομοίων όρων} Αναγωγή ομοίων όρων ονομάζεται η διαδικασία με την οποία απλοποιούμε μια αλγεβρική παράσταση, προσθέτοντας τους όμοιους όρους μεταξύ τους.\\\\
\methodologia
\Methodologia{Αλλαγή Μεταβλητήσ} Εστω $ P(x)=a_\nu x^\nu+a_{\nu-1}x^{\nu-1}+\ldots+a_1x+a_0 $ ένα πολυώνυμο. Στη θέση της αρχικής μεταβλητής μπορούμε να βάλουμε οποιαδήποτε άλλη μεταλητή ή αλγεβρική παράσταση αντικαθιστώντας την όπου αυτή υπάρχει μέσα στο πολυώνυμο.\\Για παράδειγμα, το πολυώνυμο $ P $ με μεταβλητή $ t $ θα είναι : \[ P(t)=a_\nu t^\nu+a_{\nu-1}t^{\nu-1}+\ldots+a_1t+a_0 \] ενω με μεταβλητή την παράσταση $ x^2-2x $ θα είναι : \[ P\left( x^2-2x\right) =a_\nu \left( x^2-2x\right) ^\nu+a_{\nu-1}\left(x^2-2x\right) ^{\nu-1}+\ldots+a_1\left(x^2-2x\right) +a_0 \]
\Methodologia{Αναγωγή ομοίων όρων} Εαν σε ένα πολυώνυμο ή γενικά σε μια αλγεβρική παράσταση υπάρχουν όμοιοι όροι τότε μπορούμε να την φέρουμε σε απλούστερη μορφή προσθέτοντας τους όμοιους όρους, κάνοντας πράξεις μόνο με τους συντελεστές τους όπως είδαμε στη \textbf{\textcolor{royalblue}{Μεθοδο 1.2.1}}.\\\\
\Methodologia{Πρόσθεση - Αφαίρεση Πολυονύμων} Για να προσθέσουμε ή να αφαιρέσουμε δύο πολυώνυμα $ Α(x),B(x) $ κάνουμε πράξεις μόνο μεταξύ των όμοιων μονωνύμων δηλαδή αναγωγή ομοίων όρων.\\Στην αφαίρεση $ A(x)-B(x) $ θα πρέπει να προσέξουμε να τοποθετήσουμε το 2\textsuperscript{ο} πολυώνυμο σε μια παrένθεση ώστε απαλοίφοντας την να αλλάξουν όλα του τα πρόσημα.\\

\Methodologia{Πολλαπλασιασμόσ Πολυωνύμων}
Για να πολλαπλασιάσουμε δύο πολυώνυμα, πολλαπλασιάζουμε όλους τους όρους μεταξύ τους χρησιμοποιώντας την επιμεριστική ιδιότητα και στη συνέχεια απλοποιούμε το ανάπτυγμα κάνοντας αναγωγή ομοίων όρων.\\ Εαν τα πολυώνυμα είναι τρία ή περισσότερα τα πολλαπλασιάζουμε ανά ζεύγη.

\section{Εξισώσεις 1\textsuperscript{ου} βαθμού}
\Orismos{Εξίσωση 1\textsuperscript{\MakeLowercase{ου}} βαθμού}
Εξίσωση 1\textsuperscript{ου} βαθμού με έναν άγνωστο ονομάζεται κάθε πολυωνυμική εξίσωση 1\textsuperscript{ου} βαθμού. Είναι της μορφής : \[ ax+\beta=0 \]
\begin{itemize}[itemsep=0mm]
\item Κάθε αριθμός που επαληθεύει μια εξίσωση ονομάζεται \textbf{λύση} της.
\item Η διαδικασία με την οποία βρίσκουμε τη λύση μιας εξίσωσης ονομάζεται \textbf{επίλυση}.
\item Αν ο συντελεστής της μεταβλητής  είναι διάφορος του 0 τότε η εξίσωση έχει μοναδική λύση την .
\item Εαν μια εξίσωση έχει λύσεις όλους τους πραγματικούς αριθμούς ονομάζεται \textbf{ταυτότητα} ή \textbf{αόριστη}.
\item Εαν μια εξίσωση δεν έχει καμία λύση ονομάζεται \textbf{αδύνατη}.
\end{itemize}
