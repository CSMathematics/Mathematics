\paragraph{\large ΑΣΚΗΣΕΙΣ}\mbox{}\\
\begin{enumerate}[itemsep=5mm]
\item Να λυθούν οι παρακάτω εξισώσεις.
\begin{multicols}{2}
\begin{enumerate}[itemsep=0mm,label=\roman*.]
 \item $ 4x-3=7-2x $
 \item $ 2-3x-5=8-x-1 $
 \item $ 3(1-x)+5=2-(x-4) $
 \item $ -4(3-x)+2(x-1)=5 $
\end{enumerate}
\end{multicols}
\item Να λυθούν οι παρακάτω εξισώσεις.
\begin{multicols}{2}
\begin{enumerate}[itemsep=3mm,label=\roman*.]
 \item $ \dfrac{x-2}{2}-\dfrac{x}{3}=4 $
 \item $ 2-\dfrac{1-2x}{5}=x+\dfrac{3}{4} $
 \item $ \dfrac{2(x+2)}{7}+\dfrac{2-x}{3}=1 $
 \item $ -\dfrac{1}{2}+\dfrac{3x-2}{4}=2 $
\end{enumerate}
\end{multicols}
\item Να λυθούν οι παρακάτω ανισώσεις.
\begin{multicols}{2}
\begin{enumerate}[itemsep=0mm,label=\roman*.]
 \item $ 3x-5>7-x $
 \item $ 2+3x<9-5x+1 $
 \item $ 2(3-x)+4\leq1-4(x+1) $
 \item $ 3(x+1)-4(2x-1)\geq 9-x $
\end{enumerate}
\end{multicols}
\item Να βρεθούν οι κοινές λύσεις των παρακάτω ανισώσεων.
\begin{center}
	$ \dfrac{2+x}{4}+\dfrac{3x}{5}>2 \;\;$ και $\;\; \dfrac{2x}{3}+x\leq3-\dfrac{7x}{2} $
\end{center}
\item Να υπολογίσεις τις τιμές των παρακάτω παραστάσεων
\begin{multicols}{2}
	\begin{enumerate}[itemsep=0mm,label=\roman*.]
		\item $ \SQRT{2+\!\sqrt{7+\!\sqrt{4}}} $
		\item $ \SQRT{1+\!2\sqrt{22-\!2\sqrt{9}}} $
	\end{enumerate}
\end{multicols}
\item Να χαρακτιρίσεις τους παρακάτω αριθμούς ώς \textbf{ρητούς} ή \textbf{άρρητους}.
\[ 3\;,\;\sqrt{2}\;,\;4{,}5\;,\;-\dfrac{5}{7}\;,\;\sqrt{8}\;,\;\dfrac{1}{\sqrt{3}}\;,\;7^2\;,\;\sqrt{14^2}\;,\;\sqrt{9}\;,\;\sqrt{7}\;,\;-\sqrt{1,21} \]
\item Να συμπληρωθούν οι παρακάτω πίνακες τιμών για κάθε συνάρτηση.
\begin{multicols}{2}
	\begin{enumerate}[itemsep=3mm,label=\roman*.]
		\item 	\begin{tabular}{|c|c|c|c|c|c|}
			\hline \rule[-2ex]{0pt}{5.5ex} $x$ & -2 & 0 & 1 & 2 & 3 \\ 
			\hline \rule[-2ex]{0pt}{5.5ex} $y$ &  &  &  &  &  \\ 
			\hline 
		\end{tabular} \quad $ y=2x-7 $
		\item \begin{tabular}{|c|c|c|c|c|c|}
			\hline \rule[-2ex]{0pt}{5.5ex} $x$ & -4 & -3 & -1 & 0 & 5 \\ 
			\hline \rule[-2ex]{0pt}{5.5ex} $y$ &  &  &  &  &  \\ 
			\hline 
		\end{tabular} \quad $ y=4-x $
	\end{enumerate}
\end{multicols}
\item Δίνεται η συνάρτηση $ y=(4\lambda-5)x+3 $. Να βρεθεί η τιμή του $ \lambda $ ώστε το σημείο $ Α(2,-1) $ να είναι σημείο της γραφικής παράστασης της συνάρτησης.
\item Να σχεδιάσεις τις γραφικές παραστάσεις των παρακάτω συναρτήσεων.
\begin{multicols}{2}
	\begin{enumerate}[itemsep=0mm,label=\roman*.]
		\item $ y=4x-3 $
		\item $ y=x^2+1 $
	\end{enumerate}
\end{multicols}
\item Ένα αυτοκίνητο κινείται με σταθερή ταχύτητα $ v $ με μονάδα μέτρησης $ km/h $. Αν γνωρίζουμε οτι τη χρονική στιγμή $ t=4h $ το αυτοκίνητο έχει διανύσει απόσταση $ S=320km $ τότε
\begin{enumerate}[itemsep=0mm,label=\roman*.]
	\item Να βρεθεί η ταχύτητα του αυτοκινήτου.
	\item Να γραφτεί η απόσταση $ S $ ως συνάρτηση του χρόνου $ t $.
	\item Να σχεδιαστεί η γραφική παράσταση της συνάρτησης της απόστασης $ S $.
\end{enumerate}
\end{enumerate}
