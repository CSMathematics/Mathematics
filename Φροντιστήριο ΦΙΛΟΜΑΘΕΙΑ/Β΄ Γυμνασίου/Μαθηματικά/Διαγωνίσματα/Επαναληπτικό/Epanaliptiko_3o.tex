\begin{enumerate}[label=\bf\textcolor{black}{{\large \arabic*.}},
itemsep=5mm]
\item \textbf{Να απαντήσεις στις παρακάτω ερωτήσεις.}
\begin{enumerate}[label=\roman*.]
\item Ποιά γωνία ονομάζεται εγγεγραμμένη?
\item Ποιά γωνία ονομάζεται επίκεντρη?
\item Ποιά σχέση έχει μια εγγεγραμμένη γωνία με το αντίστοιχο τόξο της?
Πότε ένα πολύγωνο ονομάζεται κανόνικό?
\item Από ποιόν τύπο δίνεται η κεντρική γωνία $ \omega $ ενός κανονικού πολυγώνου?
\item Από ποιόν τύπο δίνεται η γωνία $ \varphi $ ενός κανονικού πολυγώνου?
\item Ποιά σχέση έχουν οι γωνίες $ \omega,\varphi $ ενός κανονικού πολυγώνου?
\item Από ποιόν τύπο δίνεται το μήκος $ L $ ενός κύκλου με ακτίνα $ \rho $?
\end{enumerate}
\item \textbf{Α. Να συμπληρώσεις τα παρακάτω κενά.}
\begin{enumerate}[label=\roman*.]
\item Μια εγγεγραμμένη γωνία που βαίνει σε ημικύκλιο είναι \ldots\ldots\ldots\ldots
\item Μια επίκεντρη γωνία με το αντίστοιχο τόξο της είναι μεταξύ τους \ldots\ldots\ldots\ldots
\item Η κεντρική γωνία ενός πολυγώνου με τη γωνία του είναι μεταξύ τους \ldots\ldots\ldots\ldots
\item Ο αριθμός π είναι το πηλίκο του μήκους ενός κύκλου προς τη \ldots\ldots\ldots\ldots του.
\end{enumerate}
\textbf{Β. Να χαρακτηρίσεις τις παρακάτω προτάσεις ως σωστές (Σ) ή λάνθασμένες (Λ).\\Σε όσες προτάσεις είναι λανθασμένες, να γράψεις δίπλα τη σωστή απάντηση.}
\begin{enumerate}[label=\roman*.]
\item Αν μια εγγεγραμμένη γωνία είναι $ 70\degree $ τότε το αντίστοιχο τόξο της είναι $ 70\degree $.
\item Η κεντρική γωνία ενός κανονικού εξαγώνου είναι $ 80\degree $.
\item Η γωνία ενός κανονικού δεκαγώνου είναι $ 100\degree $.
\item Μια εγγεγραμμένη γωνία που βαίνει σε ημικύκλιο είναι $ 180\degree $.
\item Η κεντρική γωνία και η γωνία ενός κανονικού πολυγώνου είναι μεταξύ τους ίσες.
\end{enumerate}
\end{enumerate}
