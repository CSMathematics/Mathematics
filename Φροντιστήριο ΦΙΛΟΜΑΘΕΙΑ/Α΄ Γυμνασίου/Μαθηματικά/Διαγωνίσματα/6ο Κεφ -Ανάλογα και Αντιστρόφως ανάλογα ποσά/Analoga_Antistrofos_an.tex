\begin{enumerate}[label=\bf\textcolor{black}{{\large \arabic*.}},

\item \textbf{Να απαντήσεις στις παρακάτω ερωτήσεις.}
\begin{enumerate}[label=\roman*.]
\item Ποιά ποσά ονομάζονται ανάλογα? (Να δώσεις τον ορισμό).
\item Ποιά ιδιότητα έχουν οι τιμές δύο αντιστρόγως ανάλογων ποσών?
\item Τι ονομάζεται ορθοκανονικό σύστημα ημιαξόνων?
\item Αν $ x,y $ είναι δύο ανάλογα ποσά, πως ονομάζεται το πηλίκο $ \frac{y}{x} $ των τιμών τους, το οποίο παραμένει σταθερό?
\end{enumerate}\monades{3}{.94}
\item \textbf{Α. Να συμπληρώσεις τα παρακάτω κενά.}
\begin{enumerate}[label=\roman*.]
\item Η ισότητα που συνδέει δύο αντιστρόφως ανάλογα ποσά $ x,y $ είναι $ \ldots\ldots\ldots\ldots $
\item Οι αριθμοί οι οποίοι μας δίνουν τη θέση ενός σημείου στο επίπεδο λέγονται $ \ldots\ldots\ldots\ldots\ldots $ του σημείου.
\item Η γραφική παράσταση μιας σχέσης $ y=a\cdot x $ που συνδέει δύο ανάλογα ποσά έχει σχήμα $ \ldots\ldots\ldots\ldots\ldots $.
\item Ο λόγος της απόστασης δύο σημείων στην εικόνα ενός αντικειμένου προς την πραγματική τους απόσταση ονομάζεται $ \ldots\ldots\ldots\ldots\ldots $
\end{enumerate}\monades{1.5}{.94}\\
\textbf{Β. Να χαρακτηρίσεις τις παρακάτω προτάσεις ως σωστές (Σ) ή λάνθασμένες (Λ).}
\begin{enumerate}[label=\roman*.]
\item Σε κάθε διατεταγμένο ζεύγος αριθμών $ (x,y) $ που αντιστοιχεί σε ένα σημείο, το $ x $ ονομάζεται τετμημμενη του σημείου.
\item Τα σημεία $ A(3,4) $ και $ B(4,3) $ ταυτίζονται.
\item Αν ένα σημείο έχει τεταγμένη 0 τότε θα βρίσκεται πάνω στον κατακόρυφο ημιάξονα $ Oy $.
\item Οι αντίστοιχες τιμές δύο ανάλογων ποσών έχουν σταθερό πηλίκο.
\item Αναλογία ονομάζεται ο λόγος δύο ποσών.
\end{enumerate}\monades{1.5}{.94}
\end{enumerate}
